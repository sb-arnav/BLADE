    ```tsx
    // src/features/identity/EditSectionDialog.tsx
    // Shared edit-with-confirm Dialog used by SoulView + CharacterBible + PersonaView.
    // @see .planning/phases/06-life-os-identity/06-PATTERNS.md §4
    // @see .planning/phases/06-life-os-identity/06-CONTEXT.md §D-153 §D-154 §D-155

    import { useEffect, useState } from 'react';
    import { Dialog, Button } from '@/design-system/primitives';
    import { useToast } from '@/lib/context/ToastContext';

    export interface EditSectionDialogProps {
      open: boolean;
      title: string;
      initialContent: string;
      onClose: () => void;
      /** Invoked with the new content; return a promise. Dialog waits on it. */
      onSave: (content: string) => Promise<void>;
      /** Optional placeholder text for the textarea. */
      placeholder?: string;
    }

    export function EditSectionDialog(props: EditSectionDialogProps) {
      const { open, title, initialContent, onClose, onSave, placeholder } = props;
      const [value, setValue] = useState(initialContent);
      const [busy, setBusy] = useState(false);
      const toast = useToast();

      // Reset value when the dialog opens with new content.
      useEffect(() => { if (open) setValue(initialContent); }, [open, initialContent]);

      const save = async () => {
        setBusy(true);
        try {
          await onSave(value);
          toast.push({ type: 'success', message: `Saved ${title}` });
          onClose();
        } catch (e) {
          toast.push({ type: 'error', message: typeof e === 'string' ? e : String(e) });
        } finally {
          setBusy(false);
        }
      };

      return (
        <Dialog open={open} onClose={onClose} title={`Edit ${title}`}>
          <textarea
            value={value}
            onChange={(e) => setValue(e.target.value)}
            placeholder={placeholder ?? `Enter ${title}...`}
            rows={12}
            className="identity-edit-textarea"
            data-testid="identity-edit-textarea"
          />
          <div className="dialog-actions" style={{ display: 'flex', gap: 'var(--sp-2)', justifyContent: 'flex-end', marginTop: 'var(--sp-3)' }}>
            <Button variant="ghost" onClick={onClose} disabled={busy}>Cancel</Button>
            <Button variant="primary" onClick={save} disabled={busy}>
              {busy ? 'Saving…' : 'Save'}
            </Button>
          </div>
        </Dialog>
      );
    }
    ```
