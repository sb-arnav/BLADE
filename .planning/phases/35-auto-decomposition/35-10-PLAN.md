```typescript
// src/features/chat/SubagentProgressBubble.tsx
import { useEffect, useState } from 'react';
import { listen } from '@tauri-apps/api/event';
import type { BladeLoopEventPayload } from '@/lib/events/payloads';

interface ActiveSubagent {
  role: string;
  status: 'running' | 'tool_call' | 'compacting' | 'verifying';
  detail?: string;
}

/**
 * Plan 35-10 — Inline in-flight sub-agent indicator.
 *
 * Subscribes to BLADE_LOOP_EVENT subagent_started/progress/complete events
 * and renders one bubble per active sub-agent above the chat composer.
 * Removes itself from the rendered list when subagent_complete fires for
 * the matching step_index.
 *
 * Claude's discretion lock (35-CONTEXT.md §Progress Visibility — YES) —
 * chat-first pivot favors inline visibility over ActivityStrip-only chips.
 *
 * Reuses existing chat-bubble styling for visual consistency.
 */
export function SubagentProgressBubble() {
  const [activeMap, setActiveMap] = useState<Map<number, ActiveSubagent>>(new Map());

  useEffect(() => {
    const unlisten = listen<{ payload: BladeLoopEventPayload }>('blade_loop_event', (event) => {
      const payload = event.payload?.payload ?? event.payload;
      if (!payload || typeof payload !== 'object' || !('kind' in payload)) return;
      const p = payload as BladeLoopEventPayload;

      setActiveMap((prev) => {
        const next = new Map(prev);
        switch (p.kind) {
          case 'subagent_started':
            next.set(p.step_index, {
              role: p.role,
              status: 'running',
            });
            break;
          case 'subagent_progress': {
            const existing = next.get(p.step_index);
            next.set(p.step_index, {
              role: existing?.role ?? 'unknown',
              status: p.status,
              detail: p.detail,
            });
            break;
          }
          case 'subagent_complete':
            next.delete(p.step_index);
            break;
          default:
            break;
        }
        return next;
      });
    });

    return () => {
      unlisten.then((fn) => fn());
    };
  }, []);

  if (activeMap.size === 0) return null;

  // Render one bubble per active sub-agent. Sort by step_index for stable order.
  const entries = Array.from(activeMap.entries()).sort(([a], [b]) => a - b);

  return (
    <div className="subagent-progress-container">
      {entries.map(([step_index, { role, status, detail }]) => (
        <div key={step_index} className="chat-bubble subagent-progress">
          <span className="subagent-label">
            Sub-agent {step_index} ({role}): {status}
            {detail ? ` · ${detail}` : ''}
          </span>
        </div>
      ))}
    </div>
  );
}
```
