    ```ts
    // src/features/agents/useAgentTimeline.ts
    // Ref-buffer + rAF-flush event timeline for AgentDetail.
    // @see .planning/phases/05-agents-knowledge/05-PATTERNS.md §2
    // @see .planning/phases/05-agents-knowledge/05-CONTEXT.md §D-125, §D-129, §D-135

    import { useCallback, useEffect, useRef, useState } from 'react';
    import { useTauriEvent, BLADE_EVENTS } from '@/lib/events';
    import type {
      AgentEventPayload, AgentStepStartedPayload, AgentStepCompletedPayload,
    } from '@/lib/events/payloads';
    import type { TimelineRow } from './types';

    const MAX_TIMELINE_ROWS = 200;

    type AnyPayload = Record<string, unknown> | null;

    export function useAgentTimeline(currentAgentId: string | null) {
      const [rows, setRows] = useState<TimelineRow[]>([]);
      const bufferRef = useRef<TimelineRow[]>([]);
      const seqRef = useRef(0);
      const rafRef = useRef<number | null>(null);

      const scheduleFlush = useCallback(() => {
        if (rafRef.current != null) return;
        rafRef.current = requestAnimationFrame(() => {
          rafRef.current = null;
          if (bufferRef.current.length === 0) return;
          const incoming = bufferRef.current;
          bufferRef.current = [];
          setRows((prev) => {
            const next = prev.concat(incoming);
            return next.length > MAX_TIMELINE_ROWS
              ? next.slice(next.length - MAX_TIMELINE_ROWS)
              : next;
          });
        });
      }, []);

      const push = useCallback((event: string, payload: AnyPayload) => {
        const agentId = String((payload?.agent_id ?? payload?.id ?? '') as string);
        if (currentAgentId && agentId && agentId !== currentAgentId) return;
        const seq = ++seqRef.current;
        bufferRef.current.push({
          seq,
          ts: Date.now(),
          event,
          agentId,
          preview: payload ? JSON.stringify(payload).slice(0, 80) : '',
        });
        scheduleFlush();
      }, [currentAgentId, scheduleFlush]);

      const clear = useCallback(() => {
        setRows([]);
        bufferRef.current = [];
        seqRef.current = 0;
      }, []);

      // 10 subscriptions (D-129 — one per event name).
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.BLADE_AGENT_EVENT,      (p) => push('blade_agent_event', p ?? null));
      useTauriEvent<AgentStepStartedPayload | null>(BLADE_EVENTS.AGENT_STEP_STARTED,(p) => push('agent_step_started', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_RESULT,      (p) => push('agent_step_result', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_RETRYING,    (p) => push('agent_step_retrying', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_TOOL_FALLBACK,(p) => push('agent_step_tool_fallback', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_PROVIDER_FALLBACK,(p) => push('agent_step_provider_fallback', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_PARTIAL,     (p) => push('agent_step_partial', p ?? null));
      useTauriEvent<AgentStepCompletedPayload | null>(BLADE_EVENTS.AGENT_STEP_COMPLETED,(p) => push('agent_step_completed', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_STEP_FAILED,      (p) => push('agent_step_failed', p ?? null));
      useTauriEvent<AgentEventPayload | null>(BLADE_EVENTS.AGENT_EVENT,            (p) => push('agent_event', p ?? null));

      useEffect(() => () => {
        if (rafRef.current != null) cancelAnimationFrame(rafRef.current);
      }, []);

      return { rows, clear };
    }
    ```

    **CRITICAL**: the hook is called exactly once per AgentDetail mount. Re-mounting resets via the `clear()` callback exposed. Do NOT call `useAgentTimeline` inside loops or conditionals — React hook rules apply.
