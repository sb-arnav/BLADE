    **SwarmViewDev.tsx** — mounts SwarmView with a mocked Swarm via `__TAURI_INVOKE_HOOK__`:

    ```tsx
    // src/features/dev/SwarmViewDev.tsx
    import { useEffect, useState } from 'react';
    import { SwarmView } from '@/features/agents/SwarmView';

    const MOCK_SWARM = {
      id: 'mock-swarm-1',
      goal: 'Mock swarm for Playwright render test',
      status: 'running',
      created_at: Date.now(),
      steps: [
        { id: 'step-a', agent_role: 'Researcher', task: 'Investigate', deps: [],                 status: 'complete' },
        { id: 'step-b', agent_role: 'Coder',      task: 'Implement',  deps: ['step-a'],          status: 'running'  },
        { id: 'step-c', agent_role: 'Reviewer',   task: 'Review',     deps: ['step-a', 'step-b'], status: 'pending'  },
      ],
    };

    export function SwarmViewDev() {
      const [ready, setReady] = useState(false);
      useEffect(() => {
        // Install an invoke hook that returns MOCK_SWARM for swarm_list / swarm_get.
        const prevHook = window.__TAURI_INVOKE_HOOK__;
        const hook = new Map<string, (args: unknown) => unknown>();
        hook.set('swarm_list',         () => [MOCK_SWARM]);
        hook.set('swarm_get',          () => MOCK_SWARM);
        hook.set('swarm_get_progress', () => ({ completed: 1, total: 3 }));
        window.__TAURI_INVOKE_HOOK__ = hook;
        setReady(true);
        return () => {
          window.__TAURI_INVOKE_HOOK__ = prevHook;
        };
      }, []);
      if (!ready) return null;
      return <SwarmView />;
    }
    ```
