    // src/features/chat/ToolApprovalDialog.tsx — Tool approval modal with 500ms delay (D-71 / CHAT-04 / CHAT-07).

    import { useEffect, useState } from 'react';
    import { Button, Dialog } from '@/design-system/primitives';
    import { useChatCtx } from './useChat';

    /**
     * Reads `toolApprovalRequest` from chat context. Approve/Deny buttons are
     * data-countdown="on" for the first 500ms after the request lands; CSS
     * keyframe `countdownFill` animates the visual + the buttons are disabled.
     *
     * The Rust emit key for the approval id is `approval_id` (commands.rs:1636);
     * we read it as `req.approval_id` (the payload type from Phase 1 declared it
     * as `request_id` but Rust truth is `approval_id` — TS payload corrected
     * here in scope).
     */
    export function ToolApprovalDialog() {
      const { toolApprovalRequest, approveTool, denyTool } = useChatCtx();
      const [unlocked, setUnlocked] = useState(false);

      useEffect(() => {
        if (!toolApprovalRequest) { setUnlocked(false); return; }
        const t = setTimeout(() => setUnlocked(true), 500);
        return () => clearTimeout(t);
      }, [toolApprovalRequest]);

      if (!toolApprovalRequest) return null;
      const req = toolApprovalRequest as unknown as {
        approval_id?: string; request_id?: string;
        name?: string; tool_name?: string;
        arguments?: unknown; args?: unknown;
        context?: string; risk?: string;
      };
      const approvalId = req.approval_id ?? req.request_id ?? '';
      const toolName = req.tool_name ?? req.name ?? 'tool';
      const argsObj = req.args ?? req.arguments ?? {};

      return (
        <Dialog open onClose={() => denyTool(approvalId)} ariaLabel={`Approve tool: ${toolName}`}>
          <div className="tool-approval">
            <h3 className="tool-approval-title">Approve tool: <code>{toolName}</code></h3>
            <pre className="tool-approval-args">{safeStringify(argsObj)}</pre>
            {req.context ? <p className="tool-approval-context">{req.context}</p> : null}
            <div className="tool-approval-actions">
              <Button
                variant="secondary"
                onClick={() => denyTool(approvalId)}
                disabled={!unlocked}
                data-countdown={unlocked ? 'off' : 'on'}
              >Deny</Button>
              <Button
                variant="primary"
                onClick={() => approveTool(approvalId)}
                disabled={!unlocked}
                data-countdown={unlocked ? 'off' : 'on'}
              >Approve</Button>
            </div>
          </div>
        </Dialog>
      );
    }

    function safeStringify(v: unknown): string {
      try { return JSON.stringify(v, null, 2); } catch { return String(v); }
    }
    ```

    // src/features/chat/CompactingIndicator.tsx — Compacting pill (D-73 / CHAT-09).

    import { useChatCtx } from './useChat';

    /**
     * Pill shows when token ratio > 0.65 (D-16 threshold) AND chat status is not idle.
     * Threshold codified in D-73; do not parameterize without revisiting D-16.
     */
    export function CompactingIndicator() {
      const { tokenRatio, status } = useChatCtx();
      if (!tokenRatio || tokenRatio.ratio <= 0.65 || status === 'idle') return null;
      const pct = Math.round(tokenRatio.ratio * 100);
      return (
        <div className="chat-compacting" role="status" aria-live="polite">
          Compacting… {pct}%
        </div>
      );
    }
    ```
