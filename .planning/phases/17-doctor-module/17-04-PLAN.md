    /// Classify a single tentacle's severity per CONTEXT.md D-07.
    ///
    /// - Red iff `status == "dead"` OR `(now - last_heartbeat) >= 24h`
    /// - Amber iff `status == "restarting"` OR `status == "unknown"` OR `(now - last_heartbeat) >= 1h`

    fn classify_tentacle(now_secs: i64, last_heartbeat: i64, status: &str) -> Severity {

        if status == "dead" || age >= 86_400 {
            Severity::Red
        } else if status == "restarting" || status == "unknown" || age >= 3_600 {
            Severity::Amber
        } else {
            Severity::Green
        }
    }

    fn compute_tentacle_signal() -> Result<DoctorSignal, String> {
        let now_secs = chrono::Utc::now().timestamp();
        let now_ms = now_secs * 1000;

        // Supervisor-registered tentacles (perception, screen_timeline, godmode,
        // learning_engine, homeostasis, hive — see RESEARCH § D2)

                .into_iter()

                        "kind": "supervised",
                        "status": svc.status,
                        "last_heartbeat_unix_secs": svc.last_heartbeat,
                        "age_secs": now_secs.saturating_sub(svc.last_heartbeat),
                        "crash_count": svc.crash_count,
                    });
                    (svc.name, sev, payload)
                })
                .collect();

        // MCP integrations (gmail/calendar/slack/github via integration_bridge)
        let mcp_rows: Vec<(String, Severity, serde_json::Value)> =
            crate::integration_bridge::get_per_service_last_poll()
                .into_iter()
                .filter(|(_, _, enabled)| *enabled) // ignore disabled integrations
                .map(|(name, last_poll, enabled)| {
                    // last_poll is unix seconds; status="running" assumption since
                    // integration_bridge has no per-service status field
                    let status = if last_poll == 0 { "unknown".to_string() } else { "running".to_string() };
                    let sev = classify_tentacle(now_secs, last_poll, &status);
                    let payload = serde_json::json!({
                        "name": name,
                        "kind": "mcp_integration",
                        "status": status,
                        "last_poll_unix_secs": last_poll,
                        "age_secs": now_secs.saturating_sub(last_poll),
                        "enabled": enabled,
                    });
                    (name, sev, payload)
                })
                .collect();

        let mut all_rows: Vec<(String, Severity, serde_json::Value)> = supervisor_rows;
        all_rows.extend(mcp_rows);

        // Worst-of severity rollup
        let severity = all_rows
            .iter()
            .map(|(_, sev, _)| *sev)
            .max_by_key(|sev| match sev {
                Severity::Green => 0,
                Severity::Amber => 1,
                Severity::Red => 2,
            })
            .unwrap_or(Severity::Green);

        let payload = serde_json::json!({
            "total_tentacles": all_rows.len(),
            "supervised_count": all_rows.iter().filter(|(_, _, p)| p.get("kind").and_then(|k| k.as_str()) == Some("supervised")).count(),
            "mcp_count": all_rows.iter().filter(|(_, _, p)| p.get("kind").and_then(|k| k.as_str()) == Some("mcp_integration")).count(),
            "tentacles": all_rows.iter().map(|(name, sev, p)| serde_json::json!({
                "name": name,
                "severity": format!("{:?}", sev).to_lowercase(),
                "details": p,
            })).collect::<Vec<_>>(),
        });

        Ok(DoctorSignal {
            class: SignalClass::TentacleHealth,
            severity,
            payload,
            last_changed_at: now_ms,
            suggested_fix: suggested_fix(SignalClass::TentacleHealth, severity).to_string(),
        })
    }

    (2) **`fn check_migration_ledger() -> (bool, String)`** — shells out to `scripts/verify-migration-ledger.mjs`. Returns `(drift_detected, note)`:

    ```rust
    /// Run `node scripts/verify-migration-ledger.mjs` and return (drift, note).
    /// Exit code 0 → no drift; 1 → drift; anything else → treat as no-drift
    /// with note (graceful degradation if Node missing or script errored).
    fn check_migration_ledger() -> (bool, String) {
        let manifest_dir = std::path::Path::new(env!("CARGO_MANIFEST_DIR"));
        let repo_root = manifest_dir.parent().expect("CARGO_MANIFEST_DIR has parent");
        let script = repo_root.join("scripts").join("verify-migration-ledger.mjs");

        if !script.exists() {
            return (false, format!("script missing: {}", script.to_string_lossy()));
        }

        let output = std::process::Command::new("node")
            .arg(&script)
            .current_dir(repo_root)
            .output();

        match output {
            Ok(out) => {
                let code = out.status.code().unwrap_or(-1);
                match code {
                    0 => (false, "ledger clean".to_string()),
                    1 => (true, "ledger drift detected".to_string()),
                    other => (false, format!("ledger script exit {} (treated as no-drift)", other)),
                }
            }
            Err(e) => (false, format!("could not run node: {} (treated as no-drift)", e)),
        }
    }
    ```
