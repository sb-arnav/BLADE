```rust
/// Phase 33 / LOOP-05 — fast-path identity supplement.
///
/// Returns ONLY the always-keep core (identity tone, persona name, current
/// date/time, active tool list — codified by Phase 32-03 as the "small core
/// remains unconditional" decision in 32-CONTEXT.md). The full
/// character_bible / safety / hormones gates remain slow-path only.
///
/// Called from commands.rs:1441 fast-path branch BEFORE the streaming call.
/// Output is injected as ConversationMessage::System(...) so the provider
/// sees identity context even when the tool loop is bypassed.
///
/// CTX-07 fallback discipline: callers wrap this in catch_unwind. On panic,
/// the wrapped result is None and the fast path falls back to no supplement
/// (legacy behavior — preserves the v1.1 lesson that smart-path failures
/// must never break the chat).
///
/// Returns empty string when config.r#loop.smart_loop_enabled = false
/// (legacy fast-path behavior — verbatim).
///
/// Cost is ~1k tokens worst case; runs once per fast-path turn. Hard upper
/// bound 2k tokens enforced via test phase33_loop_05_supplement_capped_under_2k_tokens.
pub fn build_fast_path_supplement(
    config: &crate::config::BladeConfig,
    last_user_text: &str,
) -> String {
    if !config.r#loop.smart_loop_enabled {
        return String::new();
    }

    let _ = last_user_text;  // currently unused; kept in signature for future scoring hooks

    let persona_name = if config.user_name.is_empty() {
        "BLADE"
    } else {
        // Persona name lookup — match the slow-path source of truth. If brain.rs
        // has a `current_persona_name(&config)` helper, call it. Otherwise
        // hardcode "BLADE" — the user's chosen persona is consumed by the slow
        // path; for the fast path the identity name suffices.
        "BLADE"
    };

    let now = chrono::Utc::now();
    let date_iso = now.format("%Y-%m-%d %H:%M UTC").to_string();

    let mut out = String::new();
    out.push_str("You are ");
    out.push_str(persona_name);
    out.push_str(" — JARVIS-level desktop AI agent for ");
    if !config.user_name.is_empty() {
        out.push_str(&config.user_name);
    } else {
        out.push_str("the user");
    }
    out.push_str(". Be direct, candid, and act as a peer rather than a deferential assistant. ");
    out.push_str(&format!("Current time: {}. ", date_iso));

    // Active tool list — names only, terse, so the model knows what it CAN do.
    // Slow path provides full schemas; fast path is short-conversational so
    // names alone are sufficient context.
    out.push_str("\n\nAvailable capabilities (this turn does not need a tool call): ");
    out.push_str("native tools (file IO, bash, search, clipboard, browser, system control), ");
    if config.r#loop.smart_loop_enabled {
        // (Sanity branch — the smart-off path returned early above. This
        // condition is always true here. The branch exists so future edits
        // adding smart-only capabilities can be co-located.)
    }
    out.push_str("voice, vision, and integrations (Gmail/Calendar/Slack/GitHub when configured).");

    out
}
```

(The exact wording of the supplement is the planner's discretion — what matters is the contract: non-empty when smart-on, empty when smart-off, contains date + identity, < 2000 tokens. The execution agent can adjust the prose to match the project's voice without breaking tests.)
