```ts
test('dashboard first paint ≤ 250ms (loose CI target)', async ({ page }) => {
  await page.goto('/#/dashboard');
  const fp = await page.evaluate(() => {
    const entries = performance.getEntriesByType('paint');
    const fcp = entries.find((e) => e.name === 'first-contentful-paint');
    return fcp ? fcp.startTime : null;
  });
  expect(fp).not.toBeNull();
  expect(fp!).toBeLessThan(250);
});
```

Perf-chat-stream.spec.ts pattern (uses __TAURI_EMIT__ to simulate stream):

```ts
test('chat stream max render frame ≤ 20ms at 50 tok/sec', async ({ page }) => {
  await page.goto('/#/chat');
  // emit 50 blade_message_chunk events at 20ms intervals; measure render frame time via PerformanceObserver
  const frameTimes: number[] = await page.evaluate(async () => {
    const times: number[] = [];
    const observer = new PerformanceObserver((list) => {
      for (const e of list.getEntries()) times.push(e.duration);
    });
    observer.observe({ entryTypes: ['measure'] });
    for (let i = 0; i < 50; i++) {
      (window as any).__TAURI_EMIT__?.('blade_message_chunk', { content: `tok${i}` });
      await new Promise((r) => setTimeout(r, 20));
    }
    observer.disconnect();
    return times;
  });
  const maxFrame = Math.max(...frameTimes, 0);
  expect(maxFrame).toBeLessThan(20);
});
```
