    ```tsx
    import { useEffect, useRef } from 'react';

    export type OrbPhase = 'idle' | 'listening' | 'thinking' | 'speaking';

    interface OrbConfig { ringSpeed: number; amp: number; alpha: number; scale: number }

    function configFor(phase: OrbPhase, level: number, t: number): OrbConfig {
      switch (phase) {
        case 'idle':      return { ringSpeed: 0.6, amp: 0.35,              alpha: 0.40,              scale: 1.00 };
        case 'listening': return { ringSpeed: 0.9, amp: 0.5 + level * 0.7, alpha: 0.58 + level * 0.28, scale: 1 + level * 0.12 };
        case 'thinking':  return { ringSpeed: 0.6, amp: 0.35,              alpha: 0.40,              scale: 1.00 };
        case 'speaking':  return { ringSpeed: 1.4, amp: 0.95,              alpha: 0.72,              scale: 1 + 0.06 * Math.sin(t * 6) };
      }
    }

    /**
     * Writes OpenClaw-derived CSS vars to the orb DOM node every animation frame.
     * React state is NOT updated per frame — vars are written directly via setProperty.
     * Math locked per D-08 / RECOVERY_LOG §2.3. Do not deviate.
     */
    export function useOrbPhase(
      phase: OrbPhase,
      orbRef: React.RefObject<HTMLDivElement>,
      micRmsRef: React.MutableRefObject<number>,
    ) {
      const smoothedRef = useRef(0);
      const rafRef = useRef<number | null>(null);

      useEffect(() => {
        const start = performance.now();
        const tick = (t: number) => {
          const el = orbRef.current;
          if (!el) { rafRef.current = requestAnimationFrame(tick); return; }
          const raw = Math.max(0, Math.min(1, micRmsRef.current));
          smoothedRef.current = smoothedRef.current * 0.45 + raw * 0.55;  // locked EMA
          const level = smoothedRef.current;
          const cfg = configFor(phase, level, (t - start) / 1000);
          el.style.setProperty('--ring-speed', String(cfg.ringSpeed));
          el.style.setProperty('--amp', cfg.amp.toFixed(4));
          el.style.setProperty('--alpha', cfg.alpha.toFixed(4));
          el.style.setProperty('--orb-scale', cfg.scale.toFixed(4));
          rafRef.current = requestAnimationFrame(tick);
        };
        rafRef.current = requestAnimationFrame(tick);
        return () => { if (rafRef.current != null) cancelAnimationFrame(rafRef.current); };
      }, [phase]);
    }
    ```

    Sub-task 1b — Create `src/features/voice-orb/useMicRms.ts` per Pattern §7. Key details:
      - `navigator.mediaDevices.getUserMedia({audio: true})` acquires stream.
      - `AudioContext().createMediaStreamSource(stream).connect(analyser)` sets up analyser.
      - `analyser.fftSize = 2048`.
      - rAF loop calls `analyser.getFloatTimeDomainData(data)`, computes RMS, writes to `micRmsRef.current` (scaled by ~3x since typical speech is 0.05-0.3 raw RMS; clamp to 1).
      - On error (permission denied, no device), sets `micError` state + returns safely. Error surfaces to VoiceOrb consumers.
      - `releaseMic()` stops all tracks, closes AudioContext, nulls refs, zeroes `micRmsRef`.

    ```tsx
    import { useEffect, useRef, useState } from 'react';
    import { VoiceOrb } from './VoiceOrb';
    import type { OrbPhase } from './useOrbPhase';
    import { useMicRms } from './useMicRms';
    import { BLADE_EVENTS, useTauriEvent } from '@/lib/events';
    import type {
      VoiceConversationListeningPayload,
      VoiceConversationThinkingPayload,
      VoiceConversationSpeakingPayload,
      VoiceConversationEndedPayload,
      WakeWordDetectedPayload,
    } from '@/lib/events';
    import { invokeTyped } from '@/lib/tauri/_base';
    import { usePrefs } from '@/hooks/usePrefs';

    type Corner = 'bottom-right' | 'bottom-left' | 'top-right' | 'top-left';
    const DEFAULT_CORNER: Corner = 'bottom-right';

    export function VoiceOrbWindow() {
      const [phase, setPhase] = useState<OrbPhase>('idle');
      const { prefs, setPref } = usePrefs();
      const corner = (prefs['voice_orb.corner'] as Corner | undefined) ?? DEFAULT_CORNER;
      const { micRmsRef, acquireMic, releaseMic, micError } = useMicRms();
      const ignoreWakeUntilRef = useRef<number>(0);

      useTauriEvent<VoiceConversationListeningPayload>(BLADE_EVENTS.VOICE_CONVERSATION_LISTENING, () => {
        setPhase('listening');
        acquireMic();
      });
      useTauriEvent<VoiceConversationThinkingPayload>(BLADE_EVENTS.VOICE_CONVERSATION_THINKING, () => {
        setPhase('thinking'); releaseMic();
      });
      useTauriEvent<VoiceConversationSpeakingPayload>(BLADE_EVENTS.VOICE_CONVERSATION_SPEAKING, () => {
        setPhase('speaking'); releaseMic();
      });
      useTauriEvent<VoiceConversationEndedPayload>(BLADE_EVENTS.VOICE_CONVERSATION_ENDED, () => {
        setPhase('idle'); releaseMic();
        // 2s wake-word ignore window — prevents self-trigger via TTS tail
        ignoreWakeUntilRef.current = Date.now() + 2000;
      });
      useTauriEvent<WakeWordDetectedPayload>(BLADE_EVENTS.WAKE_WORD_DETECTED, () => {
        if (Date.now() < ignoreWakeUntilRef.current) return;
        if (phase === 'idle') {
          invokeTyped<void>('start_voice_conversation').catch(() => { /* ignore: already active */ });
        }
      });

      // Drag handling — snap to nearest quadrant on release, persist via usePrefs
      useEffect(() => {
        let isDown = false; let startX = 0, startY = 0;
        const onDown = (e: MouseEvent) => { isDown = true; startX = e.clientX; startY = e.clientY; };
        const onMove = (e: MouseEvent) => {
          if (!isDown) return;
          // translate visually during drag (CSS transform is already consuming --orb-scale;
          // use a wrapper data-drag-offset attribute to avoid clobbering scale transform)
        };
        const onUp = (e: MouseEvent) => {
          if (!isDown) return;
          isDown = false;
          const midX = window.innerWidth / 2;
          const midY = window.innerHeight / 2;
          const next: Corner =
            e.clientX < midX && e.clientY < midY ? 'top-left' :
            e.clientX >= midX && e.clientY < midY ? 'top-right' :
            e.clientX < midX && e.clientY >= midY ? 'bottom-left' : 'bottom-right';
          setPref('voice_orb.corner', next);
        };
        window.addEventListener('mousedown', onDown);
        window.addEventListener('mousemove', onMove);
        window.addEventListener('mouseup', onUp);
        return () => {
          window.removeEventListener('mousedown', onDown);
          window.removeEventListener('mousemove', onMove);
          window.removeEventListener('mouseup', onUp);
        };
      }, [setPref]);

      return (
        <div className={`voice-orb-window corner-${corner}`} data-corner={corner}>
          <VoiceOrb phase={phase} micRmsRef={micRmsRef} />
          {micError && <div className="orb-mic-error" role="alert">{micError}</div>}
        </div>
      );
    }
    ```
