    ```bash
    #!/usr/bin/env bash
    # Phase 4 Plan 04-07 — D-09 regression guard.
    # Fails CI if any cursor: property appears in src/features/ghost/** or in the ghost window bootstrap.
    set -euo pipefail

    ROOT_DIR="$(cd "$(dirname "$0")/.." && pwd)"
    FILES="$ROOT_DIR/src/features/ghost"

    MATCHES=$(grep -rE "cursor *:" "$FILES" "$BOOTSTRAP" 2>/dev/null | grep -v '^Binary' || true)

    if [ -n "$MATCHES" ]; then
    ```

    ```bash
    #!/usr/bin/env bash
    # Phase 4 Plan 04-07 — Overlay bootstrap regression guard.
    # Fails if any of the 4 Phase 4 overlay main.tsx files still contains the
    # Phase 1 placeholder string "Phase 1 bootstrap" — which would mean the
    # window was reverted to a stub.
    set -euo pipefail

    ROOT_DIR="$(cd "$(dirname "$0")/.." && pwd)"
    declare -a WINS=(quickask overlay hud ghost)
    FAIL=0

    for w in "${WINS[@]}"; do
      F="$ROOT_DIR/src/windows/$w/main.tsx"
      if [ ! -f "$F" ]; then
        echo "[verify:overlay-bootstraps] FAIL — missing $F"; FAIL=1; continue;
      fi
      if grep -q "Phase 1 bootstrap" "$F"; then
        echo "[verify:overlay-bootstraps] FAIL — $F still contains 'Phase 1 bootstrap' placeholder";
        FAIL=1;
      fi
    done

    if [ "$FAIL" = "1" ]; then exit 1; fi
    echo "[verify:overlay-bootstraps] PASS — all 4 overlay bootstraps mount real components."
    ```
