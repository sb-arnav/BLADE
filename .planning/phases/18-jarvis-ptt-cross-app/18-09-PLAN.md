//! jarvis_dispatch.rs body — consent gate + write scope + 3-tier dispatch + activity log emission.
//!
//! Wave 3 simplification: NeedsPrompt path emits consent_request and returns NoConsent
//! immediately. The frontend (Plan 17) opens ConsentDialog, persists allow_always or denied,
//! then RE-INVOKES jarvis_dispatch_action. "Allow once" is treated as a one-shot dispatch
//! WITHOUT consent_set_decision (per RESEARCH Open Q1) — deferred to v1.3 in this plan.

use serde::{Deserialize, Serialize};
use tauri::{AppHandle, Emitter};
use crate::intent_router::IntentClass;
use crate::consent::{consent_check, ConsentVerdict};
use crate::ecosystem;

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "kind", rename_all = "snake_case")]
pub enum DispatchResult {
    Executed { service: String, payload: serde_json::Value },
    NoConsent,
    HardFailedNoCreds { service: String, suggestion: String },
    NotApplicable,
}

const D17_INTENT_LABEL: &str = "action_required";

/// Emit one ActivityStrip line per dispatch outcome (D-17 LOCKED format).
fn emit_jarvis_activity(app: &AppHandle, intent_class: &str, target_service: &str, outcome: &str) {
    let line = crate::safe_slice(
        &format!("[JARVIS] {}: {} → {}", intent_class, target_service, outcome),
        200
    );
    let _ = app.emit_to("main", "blade_activity_log", serde_json::json!({
        "module":        "jarvis",
        "action":        outcome,
        "human_summary": line,
        "payload_id":    None::<String>,
        "timestamp":     std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .map(|d| d.as_secs() as i64)
            .unwrap_or(0),
    }));
}

/// Emit a consent_request event so ChatPanel opens ConsentDialog.
/// Plan 17 frontend listens; user decision triggers re-invocation of jarvis_dispatch_action.
fn emit_consent_request(
    app: &AppHandle,
    intent_class: &str,
    target_service: &str,
    action_verb: &str,
    content_preview: &str,
    request_id: &str,
) {
    let _ = app.emit_to("main", "consent_request", serde_json::json!({
        "intent_class":    intent_class,
        "target_service":  target_service,
        "action_verb":     action_verb,
        "content_preview": crate::safe_slice(content_preview, 200),
        "request_id":      request_id,
    }));
}

/// Tier 1: route to a native tentacle if one exists for this service.

                crate::tentacles::slack_outbound::slack_outbound_post_message(app.clone(), channel, text).await
                    .map(|r| serde_json::to_value(r).unwrap_or_default())

/// Tier 2: MCP fallback for services with no native tentacle.
async fn try_mcp_tool(
    service: &str,
    action: &str,
    args: &serde_json::Value,
    _app: &AppHandle,
) -> Option<Result<serde_json::Value, String>> {
    let manager = crate::mcp::manager();
    let qualified = format!("mcp__{}_{}", service, action);
    if manager.has_tool(&qualified).await {
        Some(
            manager.call_tool(&qualified, args.clone()).await
                .map_err(|e| format!("[jarvis_dispatch] MCP fallback failed: {e}"))
        )
    } else {
        None
    }
}

#[tauri::command]
pub async fn jarvis_dispatch_action(
    app: tauri::AppHandle,
    intent: IntentClass,
) -> Result<DispatchResult, String> {
    match intent {
        IntentClass::ChatOnly => Ok(DispatchResult::NotApplicable),
        IntentClass::ActionRequired { service, action } => {
            // GATE 1: consent (T-18-01 mitigation)
            match consent_check(D17_INTENT_LABEL, &service) {
                ConsentVerdict::Deny => {
                    emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "denied");
                    return Ok(DispatchResult::NoConsent);
                }
                ConsentVerdict::NeedsPrompt => {
                    // Emit consent_request; frontend opens dialog. Caller (commands.rs / Plan 14)
                    // re-invokes after user decision is persisted.
                    let request_id = uuid::Uuid::new_v4().to_string();
                    let action_verb = format!("{} on {}", action, service);
                    let content_preview = format!("(args: {})", crate::safe_slice(
                        &serde_json::to_string(&serde_json::json!({"service": service, "action": action})).unwrap_or_default(),
                        180
                    ));
                    emit_consent_request(&app, D17_INTENT_LABEL, &service, &action_verb, &content_preview, &request_id);
                    emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "denied"); // pending → treat as denied for now
                    return Ok(DispatchResult::NoConsent);
                }
                ConsentVerdict::Allow => {
                    // proceed below
                }
            }

            // GATE 2: WriteScope acquisition (T-18-02 mitigation)
            let _scope = ecosystem::grant_write_window(&service, 30);

            // Tier 1: native tentacle
            // For Wave 3, the args are not passed by Plan 14 yet; we use an empty object.
            // Plan 14 will populate `args` from the LLM's tool-call payload. Here we honor
            // the contract: if args are empty, the outbound returns Err which we route to HardFailedNoCreds.
            let args = serde_json::json!({});
            if let Some(result) = try_native_tentacle(&service, &action, &args, &app).await {
                match result {
                    Ok(payload) => {
                        emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "executed");
                        return Ok(DispatchResult::Executed { service, payload });
                    }
                    Err(e) if e.contains("Connect via Integrations tab") => {
                        emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "hard_refused");
                        return Ok(DispatchResult::HardFailedNoCreds {
                            service,
                            suggestion: e,
                        });
                    }
                    Err(e) => {
                        emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "hard_refused");
                        return Err(e);
                    }
                }
            }

            // Tier 2: MCP fallback
            if let Some(result) = try_mcp_tool(&service, &action, &args, &app).await {
                match result {
                    Ok(payload) => {
                        emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "executed");
                        return Ok(DispatchResult::Executed { service, payload });
                    }
                    Err(e) => {
                        emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "hard_refused");
                        return Err(e);
                    }
                }
            }

            // Tier 3: native_tools — out of scope for v1.2 (CONTEXT § Out of scope).
            // Fall through with HardFailedNoCreds.
            emit_jarvis_activity(&app, D17_INTENT_LABEL, &service, "hard_refused");
            Ok(DispatchResult::HardFailedNoCreds {
                service: service.clone(),
                suggestion: format!("[jarvis_dispatch] No native tentacle or MCP tool for {}/{}; native_tools tier deferred to v1.3.", service, action),
            })
        }
    }
}
