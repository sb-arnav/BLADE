```tsx
import type { HTMLAttributes } from 'react';
import { Badge } from '@/design-system/primitives';
import type { JarvisInterceptPayload } from '@/lib/events/payloads';

interface JarvisPillProps extends HTMLAttributes<HTMLDivElement> {
  payload: JarvisInterceptPayload | null;
  onDismiss?: () => void;
}

/**
 * Phase 18 (JARVIS-11) — inline pill rendered in MessageList on jarvis_intercept events.
 * Composes the Badge primitive; no new design tokens.
 *  - intercepting   → Badge tone="default"
 *  - installing     → Badge tone="warn"
 *  - retrying       → Badge tone="warn"
 *  - hard_refused   → Badge tone="hot" + dismiss button
 *
 * Auto-clear (non-hard_refused states) is the parent's responsibility (useChat
 * clears on chat_done). Hard-refused stays until user dismisses.
 */
export function JarvisPill({ payload, onDismiss, className = '', ...rest }: JarvisPillProps) {
  if (!payload) return null;

  const { action, capability, reason } = payload;
  const text = (() => {
    switch (action) {
      case 'intercepting': return 'Detecting capability gap…';
      case 'installing':   return `Installing ${capability ?? 'capability'}…`;
      case 'retrying':     return `Retrying with ${capability ?? 'capability'}…`;
      case 'hard_refused': return `Couldn't complete: ${reason ?? 'no capability'}`;
    }
  })();

  const tone = (() => {
    switch (action) {
      case 'intercepting': return 'default' as const;
      case 'installing':
      case 'retrying':     return 'warn' as const;
      case 'hard_refused': return 'hot' as const;
    }
  })();

  return (
    <div className={`jarvis-pill ${className}`} aria-live="polite" {...rest}>
      <Badge tone={tone}>{text}</Badge>
      {action === 'hard_refused' && onDismiss && (
        <button
          type="button"
          onClick={onDismiss}
          className="jarvis-pill__dismiss"
          aria-label="Dismiss JARVIS notification"
        >
          ×
        </button>
      )}
    </div>
  );
}
```
