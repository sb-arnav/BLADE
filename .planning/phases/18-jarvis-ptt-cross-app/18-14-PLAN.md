//! Wave 4 Plan 14 addition — args extraction for the dispatch pipeline.
//! Tier-1 heuristic only; Tier-2 LLM-fallback DEFERRED to v1.3 per 18-DEFERRAL.md.

pub type ArgsBag = serde_json::Map<String, serde_json::Value>;

/// Returns (intent, args). args is empty for ChatOnly OR when no fields parse.
/// Heuristic-first per D-04. LLM-fallback DEFERRED to v1.3 (18-DEFERRAL.md path B).
pub async fn classify_intent(message: &str) -> (IntentClass, ArgsBag) {
    let intent = classify_intent_class(message).await;
    let args = match &intent {
        IntentClass::ChatOnly => serde_json::Map::new(),
        IntentClass::ActionRequired { service, action } => extract_args(message, service, action),
    };
    (intent, args)
}

/// Internal — kept for back-compat with Plan 06 tests; new code calls classify_intent.
async fn classify_intent_class(message: &str) -> IntentClass {
    let lower = message.to_lowercase();
    if let Some((verb, service)) = match_heuristic(&lower) {
        return IntentClass::ActionRequired {
            service: service.to_string(),
            action: verb.to_string(),
        };
    }
    classify_intent_llm(message).await.unwrap_or(IntentClass::ChatOnly)
}

fn extract_args(message: &str, service: &str, action: &str) -> ArgsBag {
    use serde_json::Value;
    let mut bag = serde_json::Map::new();
    let trimmed = message.trim();
    match (service, action) {
        ("slack", _) => {
            // channel: after "#" or after "to "
            if let Some(channel) = extract_after_marker(trimmed, "#")
                .or_else(|| extract_after_marker(&trimmed.to_lowercase(), "to "))
            {
                bag.insert("channel".to_string(), Value::String(channel));
            }
            if let Some(text) = extract_quoted_or_after_colon(trimmed) {
                bag.insert("text".to_string(), Value::String(text));
            }
        }
        ("linear", _) => {
            // title: text after the verb up to "." or end; description: rest after the title
            // Pattern: "create a linear issue: TITLE" OR "create a linear issue TITLE"
            if let Some(title) = extract_quoted_or_after_colon(trimmed)
                .or_else(|| extract_after_action_verb(trimmed, &["create", "make", "open"]))
            {
                bag.insert("title".to_string(), Value::String(title.clone()));
                // description = whatever follows after the title in the message; for v1.2 we
                // pass the original message minus the title; cold-install demo doesn't need
                // a separate description field for the SC.
                bag.insert("description".to_string(), Value::String(trimmed.to_string()));
            }
        }
        ("github", _) => {
            // owner / repo: from "owner/repo" pattern if present
            if let Some((owner, repo)) = extract_github_owner_repo(trimmed) {
                bag.insert("owner".to_string(), Value::String(owner));
                bag.insert("repo".to_string(), Value::String(repo));
            }
            if let Some(title) = extract_quoted_or_after_colon(trimmed) {
                bag.insert("title".to_string(), Value::String(title.clone()));
                bag.insert("body".to_string(), Value::String(trimmed.to_string()));
            }
        }
        ("gmail", _) => {
            if let Some(to) = extract_after_marker(&trimmed.to_lowercase(), "to ") {
                bag.insert("to".to_string(), Value::String(to));
            }
            if let Some(subject) = extract_after_marker(&trimmed.to_lowercase(), "subject:") {
                bag.insert("subject".to_string(), Value::String(subject));
            }
            if let Some(body) = extract_quoted_or_after_colon(trimmed) {
                bag.insert("body".to_string(), Value::String(body));
            }
        }
        ("calendar", _) => {
            if let Some(title) = extract_after_marker(&trimmed.to_lowercase(), "summary of ")
                .or_else(|| extract_after_marker(&trimmed.to_lowercase(), "summarize "))
            {
                bag.insert("meeting_title".to_string(), Value::String(title));
            }
            // transcript: pass the raw message; caller (commands.rs) may override with a real
            // transcript from the meeting if available. For v1.2 cold-install demo this is fine.
            bag.insert("transcript".to_string(), Value::String(trimmed.to_string()));
        }
        _ => {}
    }
    bag
}

// ── Heuristic helpers ───────────────────────────────────────────────────────

fn extract_after_marker(s: &str, marker: &str) -> Option<String> {
    let lower = s.to_lowercase();
    let idx = lower.find(marker)?;
    let start = idx + marker.len();
    let rest = s.get(start..)?.trim();
    let end = rest.find([' ', ',', '.', '\n']).unwrap_or(rest.len());
    let candidate = rest.get(..end)?.trim();
    if candidate.is_empty() { None } else { Some(candidate.to_string()) }
}

fn extract_quoted_or_after_colon(s: &str) -> Option<String> {
    // Prefer single or double quoted content
    if let Some(start) = s.find('\'') {
        if let Some(end_rel) = s[start + 1..].find('\'') {
            let q = &s[start + 1..start + 1 + end_rel];
            if !q.is_empty() { return Some(q.to_string()); }
        }
    }
    if let Some(start) = s.find('"') {
        if let Some(end_rel) = s[start + 1..].find('"') {
            let q = &s[start + 1..start + 1 + end_rel];
            if !q.is_empty() { return Some(q.to_string()); }
        }
    }
    // Otherwise text after the FIRST colon (if any), trimmed
    if let Some(idx) = s.find(':') {
        let rest = s[idx + 1..].trim();
        if !rest.is_empty() { return Some(rest.to_string()); }
    }
    None
}

fn extract_after_action_verb(s: &str, verbs: &[&str]) -> Option<String> {
    let lower = s.to_lowercase();
    for verb in verbs {
        if let Some(idx) = lower.find(verb) {
            let after = &s[idx + verb.len()..];
            // Skip "a/an/the" articles
            let after = after.trim_start();
            let after = after
                .strip_prefix("a ").or_else(|| after.strip_prefix("an ")).or_else(|| after.strip_prefix("the "))
                .unwrap_or(after);
            // Skip "linear issue" / "github issue" / "slack message" stop words
            for stop in &["linear issue", "github issue", "slack message", "issue", "message"] {
                if let Some(stripped) = after.strip_prefix(stop) {
                    let candidate = stripped.trim_start_matches([':', ' ']).trim();
                    if !candidate.is_empty() { return Some(candidate.to_string()); }
                }
            }
            let candidate = after.trim();
            if !candidate.is_empty() { return Some(candidate.to_string()); }
        }
    }
    None
}

fn extract_github_owner_repo(s: &str) -> Option<(String, String)> {
    // Matches "owner/repo" (alphanumeric + dash + underscore + dot)
    for word in s.split_whitespace() {
        let core = word.trim_matches([',', '.', ':', ';', ')', '(']);
        if core.contains('/') && !core.starts_with('/') {
            if let Some((o, r)) = core.split_once('/') {
                if !o.is_empty() && !r.is_empty()
                    && o.chars().all(|c| c.is_alphanumeric() || c == '-' || c == '_' || c == '.')
                    && r.chars().all(|c| c.is_alphanumeric() || c == '-' || c == '_' || c == '.')
                {
                    return Some((o.to_string(), r.to_string()));
                }
            }
        }
    }
    None
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ConsentChoice {
    AllowOnce,
    AllowAlways,
    Deny,
}

static PENDING: OnceLock<Mutex<HashMap<String, oneshot::Sender<ConsentChoice>>>> = OnceLock::new();

fn pending() -> &'static Mutex<HashMap<String, oneshot::Sender<ConsentChoice>>> {
    PENDING.get_or_init(|| Mutex::new(HashMap::new()))
}

/// Request consent and AWAIT the user's decision via a tokio oneshot channel.
/// 60s timeout returns Deny by default. The frontend's ConsentDialog calls
/// consent_respond(request_id, choice) which completes the receiver.
pub async fn request_consent(
    app: &tauri::AppHandle,
    intent_class: &str,
    target_service: &str,
    action_verb: &str,
    action_kind: &str,
    content_preview: &str,
) -> ConsentChoice {
    use tauri::Emitter;
    let request_id = uuid::Uuid::new_v4().to_string();
    let (tx, rx) = oneshot::channel::<ConsentChoice>();
    if let Ok(mut map) = pending().lock() {
        map.insert(request_id.clone(), tx);
    } else {
        return ConsentChoice::Deny;
    }
    let payload = serde_json::json!({
        "intent_class":    intent_class,
        "target_service":  target_service,
        "action_verb":     action_verb,
        "action_kind":     action_kind,
        "content_preview": crate::safe_slice(content_preview, 200),
        "request_id":      request_id,
    });
    let _ = app.emit_to("main", "consent_request", payload);
    match tokio::time::timeout(std::time::Duration::from_secs(60), rx).await {
        Ok(Ok(choice)) => choice,
        Ok(Err(_recv_err)) => ConsentChoice::Deny,
        Err(_timeout) => {
            // Cleanup the pending entry on timeout
            if let Ok(mut map) = pending().lock() {
                map.remove(&request_id);
            }
            ConsentChoice::Deny
        }
    }
}

#[tauri::command]
pub fn consent_respond(request_id: String, choice: String) -> Result<(), String> {
    let parsed = match choice.as_str() {
        "allow_once"   => ConsentChoice::AllowOnce,
        "allow_always" => ConsentChoice::AllowAlways,
        "denied"       => ConsentChoice::Deny,
        other => return Err(format!("[consent_respond] invalid choice: {other}")),
    };
    let sender = pending().lock()
        .map_err(|e| format!("[consent_respond] mutex poisoned: {e}"))?
        .remove(&request_id);
    match sender {
        Some(tx) => tx.send(parsed).map_err(|_| "[consent_respond] receiver dropped".to_string()),
        None => Err(format!("[consent_respond] unknown request_id: {request_id}")),
    }
}

/// Testability seam — production path uses consent_check (default db_path).
pub fn consent_check_at(db_path: &std::path::Path, intent: &str, service: &str) -> ConsentVerdict {
    let conn = match open_db_at(db_path) {
        Ok(c) => c,
        Err(_) => return ConsentVerdict::NeedsPrompt,
    };
    let row: Result<String, _> = conn.query_row(
        "SELECT decision FROM consent_decisions WHERE intent_class = ?1 AND target_service = ?2",
        rusqlite::params![intent, service],
        |row| row.get(0),
    );
    match row {
        Ok(d) if d == "allow_always" => ConsentVerdict::Allow,
        Ok(d) if d == "denied" => ConsentVerdict::Deny,
        _ => ConsentVerdict::NeedsPrompt,
    }
}
