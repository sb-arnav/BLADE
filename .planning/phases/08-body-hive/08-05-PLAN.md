```ts
// tests/e2e/body-map.spec.ts — pattern (08-PATTERNS §7)
import { test, expect } from '@playwright/test';
import { MOCK_BODY_SUMMARY, MOCK_BODY_MAP } from './_fixtures/hive-status';

test.beforeEach(async ({ page }) => {
  await page.addInitScript(({ summary, map }) => {
    (window as any).__TAURI_INVOKE_HOOK__ = (cmd: string) => {
      if (cmd === 'body_get_summary') return summary;
      if (cmd === 'body_get_map') return map;
      if (cmd === 'body_get_system') return map.filter((m: any) => m.body_system === 'nervous');
      return null;
    };
  }, { summary: MOCK_BODY_SUMMARY, map: MOCK_BODY_MAP });
});

test('BodyMap renders system cards + click drill-in (SC-1)', async ({ page }) => {
  await page.goto('/#/body-map');
  await expect(page.getByTestId('body-map-root')).toBeVisible();
  const cards = page.locator('[data-testid^="body-system-card-"]');
  await expect(cards).toHaveCount(MOCK_BODY_SUMMARY.length);
  await cards.first().click();
  await expect(page.getByTestId('body-system-detail-root')).toBeVisible();
});
```
