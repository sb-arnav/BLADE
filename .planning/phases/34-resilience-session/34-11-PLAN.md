// src/features/sessions/SessionsView.tsx

// Phase 34 / SESS-03 + SESS-04 — Sessions list UI.

// Reuses existing list-card patterns from src/features/dashboard. NO bespoke
// design system work per CONTEXT lock §SESS-03.

import { useEffect, useState, useCallback } from 'react';
import {
  listSessions, forkSession, type SessionMeta,
} from '@/lib/tauri/sessions';

interface SessionsViewProps {
  onBack: () => void;
  onResume: (id: string) => void;
}

export function SessionsView({ onBack, onResume }: SessionsViewProps) {
  const [sessions, setSessions] = useState<SessionMeta[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [branchTarget, setBranchTarget] = useState<SessionMeta | null>(null);
  const [branchIdx, setBranchIdx] = useState<number>(1);

  const refresh = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const list = await listSessions();
      setSessions(list);
    } catch (e) {
      setError(typeof e === 'string' ? e : String(e));
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { refresh(); }, [refresh]);

  const handleBranch = async () => {
    if (!branchTarget) return;
    try {
      await forkSession(branchTarget.id, branchIdx);
      setBranchTarget(null);
      await refresh();
    } catch (e) {
      setError(typeof e === 'string' ? e : String(e));
    }
  };

  return (
    <div className="sessions-view" style={{ padding: 24 }}>
      <header style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
        <h1>Sessions ({sessions.length})</h1>
        <button onClick={onBack}>Back</button>
      </header>
      {loading && <p>Loading…</p>}
      {error && <p style={{ color: 'tomato' }}>Error: {error}</p>}
      {!loading && sessions.length === 0 && <p>No past sessions yet.</p>}
      <ul style={{ listStyle: 'none', padding: 0 }}>
        {sessions.map(s => (
          <li key={s.id} className="session-card" style={{
            padding: 12, marginBottom: 8, border: '1px solid #333', borderRadius: 8
          }}>
            <div style={{ display: 'flex', justifyContent: 'space-between' }}>
              <div>
                <strong>{new Date(s.started_at_ms).toLocaleString()}</strong>
                {s.parent && <span style={{ marginLeft: 8, fontSize: 12, opacity: 0.7 }}>
                  forked from {s.parent.slice(0, 8)}…
                </span>}
                {s.halt_reason && <span style={{
                  marginLeft: 8, padding: '2px 6px', background: '#553', borderRadius: 4, fontSize: 11
                }}>{s.halt_reason}</span>}
              </div>
              <div>
                <button onClick={() => onResume(s.id)} style={{ marginRight: 4 }}>Resume</button>
                <button onClick={() => { setBranchTarget(s); setBranchIdx(Math.min(3, s.message_count)); }}
                        style={{ marginRight: 4 }}>Branch</button>
                <button disabled title="v1.6 — auto-rotation handles overflow today">Archive</button>
              </div>
            </div>
            <div style={{ marginTop: 8, fontSize: 14 }}>{s.first_message_excerpt}</div>
            <div style={{ marginTop: 4, fontSize: 12, opacity: 0.7 }}>
              {s.message_count} messages · ~{s.approximate_tokens} tokens
            </div>
          </li>
        ))}
      </ul>
      {branchTarget && (
        <div className="branch-picker-modal" style={{
          position: 'fixed', top: '50%', left: '50%', transform: 'translate(-50%, -50%)',
          background: '#222', padding: 24, border: '1px solid #555', borderRadius: 8,
        }}>
          <h2>Branch session at message #</h2>
          <p style={{ fontSize: 12, opacity: 0.7 }}>
            Parent has {branchTarget.message_count} messages. Picking N = the new session
            contains the first N user/assistant messages.
          </p>
          <input type="number" min={1} max={branchTarget.message_count} value={branchIdx}
                 onChange={e => setBranchIdx(parseInt(e.target.value, 10) || 1)}
                 style={{ padding: 8, fontSize: 16 }} />
          <div style={{ marginTop: 12 }}>
            <button onClick={handleBranch} style={{ marginRight: 8 }}>Branch</button>
            <button onClick={() => setBranchTarget(null)}>Cancel</button>
          </div>
        </div>
      )}
    </div>
  );
}

        cap: e.payload.cap_usd,
        percent: e.payload.percent,
      });
    }
  }).then(fn => { unlistenFn = fn; });
  return () => { if (unlistenFn) unlistenFn(); };
}, []);
