//! INTEL-01 per-language symbol extraction (TS/JS, Rust, Python).
//!
//! Each language has a static s-expression query that captures function
//! definitions, type definitions, imports, and call sites. Captures are
//! walked into a flat Vec<ParsedSymbol> + Vec<ParsedEdge>; symbol_graph.rs
//! consumes this output and persists into kg_nodes/kg_edges.
//!
//! Catch_unwind discipline lives at the CALLER (symbol_graph::reindex_project) —
//! this module's parse functions return Result<_, String> for clean errors,
//! and the FORCE_PARSE_ERROR seam exists for test injection without crafting
//! malformed source files.

use std::cell::Cell;

#[derive(Debug, Clone)]
pub struct ParsedSymbol {
    pub name: String,
    pub kind: ParsedSymbolKind,
    pub line_start: u32,
    pub line_end: u32,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ParsedSymbolKind { Function, Type, Module, Constant }

#[derive(Debug, Clone)]
pub struct ParsedEdge {
    pub from_name: String,    // resolved at the symbol_graph layer
    pub to_name: String,
    pub kind: ParsedEdgeKind,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum ParsedEdgeKind { Calls, Imports, UsesType }

#[derive(Debug, Clone)]
pub struct ParsedFile {
    pub symbols: Vec<ParsedSymbol>,
    pub edges: Vec<ParsedEdge>,
}

#[cfg(test)]
thread_local! {
    pub static INTEL_FORCE_PARSE_ERROR: Cell<Option<String>> = const { Cell::new(None) };
}

#[cfg(not(test))]
thread_local! {
    pub static INTEL_FORCE_PARSE_ERROR: Cell<Option<String>> = const { Cell::new(None) };
}

fn check_force_error() -> Result<(), String> {
    INTEL_FORCE_PARSE_ERROR.with(|c| {
        if let Some(msg) = c.take() {
            // Re-set so subsequent calls in the same test see the same error.
            // (Test cleanup MUST .set(None).)
            c.set(Some(msg.clone()));
            return Err(msg);
        }
        Ok(())
    })
}

pub fn parse_typescript(content: &str) -> Result<ParsedFile, String> {
    check_force_error()?;
    let mut parser = tree_sitter::Parser::new();
    parser
        .set_language(&tree_sitter_typescript::language_typescript())
        .map_err(|e| format!("set_language ts: {e}"))?;
    run_query(&mut parser, content, TS_QUERY)
}

/// Walk the parsed tree, collect captures, build ParsedFile. Edges are not
/// scope-resolved here — symbol_graph.rs::link_calls_within_project resolves
/// edges based on file proximity + same-name match.
fn run_query(
    parser: &mut tree_sitter::Parser,
    content: &str,
    query_src: &str,
) -> Result<ParsedFile, String> {
    let tree = parser.parse(content, None).ok_or_else(|| "tree-sitter parse failed".to_string())?;
    let language = parser.language().ok_or("no language set")?;
    let query = tree_sitter::Query::new(&language, query_src)
        .map_err(|e| format!("compile query: {e}"))?;
    let mut cursor = tree_sitter::QueryCursor::new();
    let bytes = content.as_bytes();

    let mut symbols: Vec<ParsedSymbol> = Vec::new();
    let mut edges: Vec<ParsedEdge> = Vec::new();

    // Track the current enclosing-function name during cursor walk, used as
    // the from-name for Calls edges. v1 picks the "last def we saw at or
    // before this byte position" via byte-range containment scan.

    for m in cursor.matches(&query, tree.root_node(), bytes) {
        for cap in m.captures {
            let cap_name = &query.capture_names()[cap.index as usize];
            let node = cap.node;
            let text = node.utf8_text(bytes).unwrap_or("").to_string();
            let line_start = node.start_position().row as u32;
            let line_end = node.end_position().row as u32;
            match cap_name.as_str() {
                "function.def" => symbols.push(ParsedSymbol {
                    name: text,
                    kind: ParsedSymbolKind::Function,
                    line_start,
                    line_end,
                }),
                "type.def" => symbols.push(ParsedSymbol {
                    name: text,
                    kind: ParsedSymbolKind::Type,
                    line_start,
                    line_end,
                }),
                "function.call" => {
                    // resolve from_name at link-time; here we record the call site name only
                    edges.push(ParsedEdge {
                        from_name: String::new(), // resolved later
                        to_name: text,
                        kind: ParsedEdgeKind::Calls,
                    });
                }
                "import.path" => {
                    edges.push(ParsedEdge {
                        from_name: String::new(), // module-level import, resolved later
                        to_name: text.trim_matches(|c: char| c == '"' || c == '\'').to_string(),
                        kind: ParsedEdgeKind::Imports,
                    });
                }
                "type.use" => {
                    edges.push(ParsedEdge {
                        from_name: String::new(),
                        to_name: text,
                        kind: ParsedEdgeKind::UsesType,
                    });
                }
                _ => {}
            }
        }
    }

    Ok(ParsedFile { symbols, edges })
}

#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub struct SymbolNode {
    pub id: String,
    pub name: String,
    pub kind: SymbolKind,
    pub file_path: String,
    pub line_start: u32,
    pub line_end: u32,
    pub language: String,
    pub indexed_at: i64,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum SymbolKind { Function, Type, Module, Constant }

impl From<ParsedSymbolKind> for SymbolKind {
    fn from(p: ParsedSymbolKind) -> Self {
        match p {
            ParsedSymbolKind::Function => SymbolKind::Function,
            ParsedSymbolKind::Type => SymbolKind::Type,
            ParsedSymbolKind::Module => SymbolKind::Module,
            ParsedSymbolKind::Constant => SymbolKind::Constant,
        }
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ReindexStats {
    pub files_walked: u32,
    pub files_parsed: u32,
    pub files_skipped: u32,
    pub symbols_inserted: u32,
    pub edges_inserted: u32,
    pub elapsed_ms: u64,
}

pub fn symbol_id(file_path: &str, name: &str, kind: SymbolKind) -> String {
    let mut hasher = Sha256::new();
    hasher.update(format!("{file_path}::{name}::{kind:?}").as_bytes());
    let result = hasher.finalize();
    hex::encode(&result[..8])  // 16 hex chars
}

pub fn reindex_project(project_root: &Path, conn: &Connection) -> Result<ReindexStats, String> {
    let start = Instant::now();
    let mut stats = ReindexStats {
        files_walked: 0,
        files_parsed: 0,
        files_skipped: 0,
        symbols_inserted: 0,
        edges_inserted: 0,
        elapsed_ms: 0,
    };

    // Step 1: clear prior symbol rows for this project.
    let root_str = project_root.to_string_lossy().to_string();
    let like_pat = format!("%\"file_path\":\"{}/%", root_str);
    conn.execute(
        "DELETE FROM kg_nodes WHERE node_type = 'symbol' AND description LIKE ?1",
        params![like_pat],
    ).map_err(|e| format!("delete prior symbol nodes: {e}"))?;
    conn.execute(
        "DELETE FROM kg_edges WHERE relation IN ('calls','imports','uses_type','defines') AND (from_id LIKE 'sym:%' OR to_id LIKE 'sym:%')",
        [],
    ).map_err(|e| format!("delete prior symbol edges: {e}"))?;

    // Step 2: walk project root.
    let mut files: Vec<PathBuf> = Vec::new();
    walk_dir(project_root, &mut files);
    stats.files_walked = files.len() as u32;

    // Step 3-7: parse + insert.
    for file in files {
        let language = detect_language(&file);
        let language = match language {
            Some(l) => l,
            None => { stats.files_skipped += 1; continue; }
        };
        let content = match std::fs::read_to_string(&file) {
            Ok(c) if c.len() <= 1_000_000 => c,
            _ => { stats.files_skipped += 1; continue; }
        };
        let parse_result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
            match language {
                "typescript" => parse_typescript(&content),
                "rust" => parse_rust(&content),
                "python" => parse_python(&content),
                _ => Err("unsupported".to_string()),
            }
        }));
        let parsed = match parse_result {
            Ok(Ok(p)) => p,
            Ok(Err(e)) => {
                log::warn!("[INTEL-01] parse error {}: {}", file.display(), e);
                stats.files_skipped += 1;
                continue;
            }
            Err(_) => {
                log::warn!("[INTEL-01] parser panic on {}; skipping", file.display());
                stats.files_skipped += 1;
                continue;
            }
        };
        stats.files_parsed += 1;

        // Insert SymbolNodes.
        let now = chrono::Utc::now().timestamp();
        let file_str = file.to_string_lossy().to_string();
        let mut file_symbols: Vec<SymbolNode> = Vec::with_capacity(parsed.symbols.len());
        for ps in &parsed.symbols {
            let kind: SymbolKind = ps.kind.into();
            let id = symbol_id(&file_str, &ps.name, kind);
            let node = SymbolNode {
                id: format!("sym:{id}"),
                name: ps.name.clone(),
                kind,
                file_path: file_str.clone(),
                line_start: ps.line_start,
                line_end: ps.line_end,
                language: language.to_string(),
                indexed_at: now,
            };
            let payload = serde_json::to_string(&node).unwrap_or_default();
            conn.execute(
                "INSERT OR REPLACE INTO kg_nodes (id, node_type, name, description, created_at) VALUES (?1, 'symbol', ?2, ?3, ?4)",
                params![node.id, node.name, payload, now],
            ).ok();
            stats.symbols_inserted += 1;
            file_symbols.push(node);
        }

        // Resolve + insert edges (within-file only for v1; cross-file resolution v1.6+).
        for edge in parsed.edges {
            // Pick from_name = the SymbolNode whose line range encloses the edge source — but parser doesn't currently emit edge source line. Fallback: use the file's first function-kind symbol as enclosing scope (good enough for ranking).
            let from_node = file_symbols.iter().find(|s| s.kind == SymbolKind::Function).cloned();
            let to_node = file_symbols.iter().find(|s| s.name == edge.to_name).cloned();
            let (from_id, to_id) = match (&from_node, &to_node) {
                (Some(f), Some(t)) => (f.id.clone(), t.id.clone()),
                _ => continue,  // unresolved cross-file or external — drop in v1
            };
            let relation = match edge.kind {
                ParsedEdgeKind::Calls => "calls",
                ParsedEdgeKind::Imports => "imports",
                ParsedEdgeKind::UsesType => "uses_type",
            };
            conn.execute(
                "INSERT OR IGNORE INTO kg_edges (from_id, to_id, relation, weight, created_at) VALUES (?1, ?2, ?3, 1.0, ?4)",
                params![from_id, to_id, relation, now],
            ).ok();
            stats.edges_inserted += 1;
        }
    }

    stats.elapsed_ms = start.elapsed().as_millis() as u64;
    Ok(stats)
}

fn walk_dir(root: &Path, out: &mut Vec<PathBuf>) {
    let entries = match std::fs::read_dir(root) { Ok(e) => e, Err(_) => return };
    for ent in entries.flatten() {
        let p = ent.path();
        let name = p.file_name().and_then(|n| n.to_str()).unwrap_or("");
        if matches!(name, "target" | "node_modules" | ".git" | "dist" | "build" | "out") { continue; }
        if p.is_dir() {
            walk_dir(&p, out);
        } else if p.is_file() {
            out.push(p);
        }
    }
}

fn detect_language(path: &Path) -> Option<&'static str> {
    let ext = path.extension()?.to_str()?;
    match ext {
        "ts" | "tsx" | "mts" | "cts" | "js" | "jsx" | "mjs" | "cjs" => Some("typescript"),
        "rs" => Some("rust"),
        "py" => Some("python"),
        _ => None,
    }
}

        let conn = fixture_conn();
        let dir = tempfile::tempdir().unwrap();

        std::fs::write(&f, "fn hello() {}\nfn world() { hello(); }\n").unwrap();
        let stats = reindex_project(dir.path(), &conn).expect("reindex");
        assert!(stats.symbols_inserted >= 2);
        let count: i64 = conn.query_row(
            "SELECT COUNT(*) FROM kg_nodes WHERE node_type = 'symbol'",
            [], |r| r.get(0)).unwrap();
        assert_eq!(count, stats.symbols_inserted as i64);

    #[test]
    fn phase36_intel_01_reindex_is_idempotent() {
        let conn = fixture_conn();
        let dir = tempfile::tempdir().unwrap();
        let f = dir.path().join("a.py");
        std::fs::write(&f, "def x():\n    pass\n").unwrap();
        let s1 = reindex_project(dir.path(), &conn).unwrap();
        let s2 = reindex_project(dir.path(), &conn).unwrap();
        assert_eq!(s1.symbols_inserted, s2.symbols_inserted, "idempotent reindex");
    }
