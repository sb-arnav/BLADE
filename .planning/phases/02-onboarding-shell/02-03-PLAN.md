    ```tsx
    // src/design-system/shell/TitleBar.tsx — 34px main-window chrome bar (SHELL-01, D-54).
    //
    // Drag region: data-tauri-drag-region on the root; interactive children opt out.
    // Layout: 3-column grid — traffic lights (left), title + status pill (center),
    // ⌘K hint chip (right).
    //
    // No role switcher — role-switching is Phase 6 IDEN scope (D-54).
    //
    // @see .planning/phases/02-onboarding-shell/02-CONTEXT.md §D-54
    // @see .planning/phases/02-onboarding-shell/02-PATTERNS.md (no specific recipe —
    //   this is a straightforward composition; mine src.bak/TitleBar.tsx lines
    //   for traffic-light glyphs but don't import).

    import { useState } from 'react';
    import { BLADE_EVENTS, useTauriEvent } from '@/lib/events';
    import type { BladeStatusPayload } from '@/lib/events';
    import { minimizeWindow, closeWindow, toggleMaximize } from '@/lib/tauri';

    type Status = 'processing' | 'thinking' | 'idle' | 'error';

    const STATUS_LABEL: Record<Status, string> = {
      processing: 'Working',
      thinking: 'Thinking',
      idle: 'Ready',
      error: 'Error',
    };

    export function TitleBar() {
      const [status, setStatus] = useState<Status>('idle');

      useTauriEvent<BladeStatusPayload>(BLADE_EVENTS.BLADE_STATUS, (e) => {
        // Payload is the status string itself (event catalog: 'processing'|'thinking'|'idle'|'error')
        const next = e.payload as Status;
        if (next === 'processing' || next === 'thinking' || next === 'idle' || next === 'error') {
          setStatus(next);
        }
      });

      return (
        <header className="titlebar" data-tauri-drag-region role="banner" aria-label="Window chrome">
          {/* Left: traffic lights (macOS-style). On other OSes these still render;
              that's fine — Tauri abstracts close/minimize. The visual feel is one
              BLADE-native look across all 3 platforms per D-15 (single aesthetic). */}
          <div className="titlebar-traffic" data-tauri-drag-region="false">
            <button type="button" className="tlight tlight-close" aria-label="Close window"
                    onClick={() => { void closeWindow(); }}>
              <span aria-hidden="true">×</span>
            </button>
            <button type="button" className="tlight tlight-min" aria-label="Minimize window"
                    onClick={() => { void minimizeWindow(); }}>
              <span aria-hidden="true">–</span>
            </button>
            <button type="button" className="tlight tlight-max" aria-label="Toggle maximize"
                    onClick={() => { void toggleMaximize(); }}>
              <span aria-hidden="true">+</span>
            </button>
          </div>

          {/* Center: title + status pill */}
          <div className="titlebar-title" data-tauri-drag-region>
            <span className="titlebar-brand">BLADE</span>
            <span className={`titlebar-status titlebar-status-${status}`}
                  data-tauri-drag-region="false"
                  aria-live="polite">
              <span className="titlebar-status-dot" aria-hidden="true" />
              {STATUS_LABEL[status]}
            </span>
          </div>

          {/* Right: ⌘K hint. Plan 05 wires the actual keyboard capture;
              this is visual only. */}
          <div className="titlebar-hint" data-tauri-drag-region="false" aria-hidden="true">
            <kbd>⌘K</kbd>
          </div>
        </header>
      );
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User click → window control | Traffic lights call `closeWindow/minimizeWindow/toggleMaximize`; no state validation needed (user owns the window) |
| Rust `blade_status` emit → TitleBar state | Payload type narrowed at runtime via literal check; unknown strings ignored |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Tampering | Drag region spans interactive children | mitigate | `data-tauri-drag-region="false"` on traffic lights, status pill button, hint chip — user input reaches clicks, not drags |
| T-02-03-02 | Tampering | Status payload carries unexpected string | mitigate | Literal-check narrowing (`next === 'processing' || ...`); unknown values ignored |
| T-02-03-03 | Elevation | Traffic-light buttons invoke window lifecycle | accept | Tauri wrapper is the only surface; user owns the window already |
| T-02-03-04 | DoS | `blade_status` emit loop | accept | React's setState short-circuits when next==current; no infinite loop |
| T-02-03-05 | Tampering | Keyboard focus trap inside titlebar | mitigate | All interactive children have `aria-label` + native button semantics; focus-visible ring via token |
</threat_model>
