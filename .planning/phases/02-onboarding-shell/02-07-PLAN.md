    // tests/e2e/onboarding-boot.spec.ts — SC-1 + SC-2 falsification (Phase 2).
    //
    // Scenario A (SC-1): Fresh launch (config.onboarded=false) → provider picker
    //   → type Anthropic key → Test (mocked success) → deep scan (mocked progress
    //   events + completion) → persona form → Enter BLADE → shell data-gate-status
    //   transitions to 'complete', dashboard route mounts.
    //
    // Scenario B (SC-2): Returning user (config.onboarded=true,
    //   persona_onboarding_complete=true) → boot directly to dashboard; no
    //   onboarding surface visible.
    //
    // Invoke mocks: install via addInitScript before the bundle loads.

    import { test, expect, Page } from '@playwright/test';

    interface InvokeLog { cmd: string; args: unknown }

    async function installMocks(page: Page, opts: {
      onboarded: boolean;
      personaDone: boolean;
      testProviderShouldFail?: boolean;
    }) {
      await page.addInitScript((scenario) => {
        const emitted: Array<{ name: string; payload: unknown }> = [];
        (window as any).__BLADE_INVOKE_LOG = [] as InvokeLog[];
        (window as any).__BLADE_EMITTED = emitted;

        const baseConfig = {
          provider: scenario.onboarded ? 'anthropic' : 'anthropic',
          model: 'claude-sonnet-4-20250514',
          api_key: scenario.onboarded ? 'sk-abc' : '',
          onboarded: scenario.onboarded,
          persona_onboarding_complete: scenario.personaDone,
          user_name: '',
        };

        (window as any).__TAURI_INTERNALS__ = {
          invoke: async (cmd: string, args: unknown) => {
            (window as any).__BLADE_INVOKE_LOG.push({ cmd, args });
            switch (cmd) {
              case 'get_config':             return { ...baseConfig };
              case 'get_onboarding_status':  return scenario.personaDone;
              case 'test_provider':
                if (scenario.testProviderShouldFail) throw 'Invalid key';
                return 'Connection OK — claude-sonnet-4';
              case 'store_provider_key':     return null;
              case 'switch_provider':        return { ...baseConfig, provider: (args as any).provider };
              case 'set_config':             return null;
              case 'deep_scan_start': {
                // Synthesize 11 deep_scan_progress emissions, then resolve.
                const phases = ['starting','installed_apps','git_repos','ides','ai_tools','wsl_distros','ssh_keys','package_managers','docker','bookmarks','complete'];
                for (const p of phases) {
                  // Listeners registered via @tauri-apps/api/event — simulate via CustomEvent bus
                  // if harness exposes one; else poll via our own event dispatcher.
                  (window as any).__BLADE_EMITTED.push({ name: 'deep_scan_progress', payload: { phase: p, found: 0 } });
                  // Actual listener dispatch:
                  window.dispatchEvent(new CustomEvent('tauri://deep_scan_progress', { detail: { phase: p, found: 0 } }));
                }
                return { scanned_at: Date.now() };
              }
              case 'complete_onboarding':    return null;
              default:                       return null;
            }
          },
        };
      }, opts);
    }

    // Bridge: Phase 1 useTauriEvent relies on @tauri-apps/api/event listen().
    // The harness installs a lightweight listen() shim that forwards to window events.
    async function installEventShim(page: Page) {
      await page.addInitScript(() => {
        const listeners: Record<string, Array<(e: unknown) => void>> = {};
        (window as any).__TAURI_EVENT_LISTEN = async (name: string, cb: (e: any) => void) => {
          listeners[name] = listeners[name] || [];
          listeners[name].push(cb);
          const handler = (ev: Event) => cb({ event: name, payload: (ev as CustomEvent).detail });
          window.addEventListener(`tauri://${name}`, handler);
          return () => window.removeEventListener(`tauri://${name}`, handler);
        };
      });
    }

    test.describe('Phase 2 onboarding boot gate', () => {
      test('SC-1: fresh launch walks provider → key → scan → persona → shell', async ({ page }) => {
        await installEventShim(page);
        await installMocks(page, { onboarded: false, personaDone: false });
        await page.goto('/');

        // Gate should render the onboarding surface with needs_provider_key status
        await expect(page.locator('[data-gate-status="needs_provider_key"]')).toBeVisible();
        await expect(page.getByRole('heading', { name: /pick a provider/i })).toBeVisible();

        // Anthropic is default-selected; click Continue
        await page.getByRole('button', { name: /continue/i }).first().click();

        // Step 2 — API key entry
        await expect(page.getByRole('heading', { name: /paste your anthropic key/i })).toBeVisible();
        await page.getByLabel(/api key/i).fill('sk-ant-test-0000');
        await page.getByRole('button', { name: /test.*continue/i }).click();

        // Step 3 — Deep scan (mocked: resolves instantly + synthesizes 11 phases)
        await expect(page.locator('[data-route-id="onboarding"], text=Learning your machine')).toBeVisible();
        // Progress should hit 100% because 'complete' phase fires
        await expect(page.getByRole('progressbar')).toHaveAttribute('aria-valuenow', '100', { timeout: 5_000 });
        await page.getByRole('button', { name: /continue/i }).click();

        // Step 4 — Persona
        await expect(page.getByRole('heading', { name: /a few quick questions/i })).toBeVisible();
        for (let i = 0; i < 5; i++) {
          await page.locator(`#persona-${i}`).fill(i === 0 ? 'Arnav, founder' : `answer ${i}`);
        }
        await page.getByRole('button', { name: /enter blade/i }).click();

        // Shell should mount (gate flips to complete after reEvaluate)
        await expect(page.locator('[data-gate-status="complete"]')).toBeVisible({ timeout: 5_000 });
        await expect(page.locator('[data-route-id="dashboard"]')).toBeVisible();
      });

      test('SC-2: returning user boots straight to dashboard', async ({ page }) => {

    // tests/e2e/shell.spec.ts — SC-3 + SC-4 falsification (Phase 2).
    //
    // SC-3: ⌘K opens palette, fuzzy filter on route label, Enter navigates, Esc closes,
    //        NO App.tsx edit required (test the registry-derived behavior).
    // SC-4: Backend event → toast appears → auto-dismisses without blocking interaction.

    import { test, expect, Page } from '@playwright/test';

    async function bootAsReturning(page: Page) {
      await page.addInitScript(() => {
        const cfg = {
          provider: 'anthropic', model: 'claude-sonnet-4-20250514', api_key: 'sk-test',
          onboarded: true, persona_onboarding_complete: true, user_name: 'Arnav',
        };
        (window as any).__TAURI_INTERNALS__ = {
          invoke: async (cmd: string) => {
            if (cmd === 'get_config') return { ...cfg };
            if (cmd === 'get_onboarding_status') return true;
            return null;
          },
        };
        (window as any).__TAURI_EVENT_LISTEN = async (name: string, cb: (e: any) => void) => {
          const handler = (ev: Event) => cb({ event: name, payload: (ev as CustomEvent).detail });
          window.addEventListener(`tauri://${name}`, handler);
          return () => window.removeEventListener(`tauri://${name}`, handler);
        };
      });
      await page.goto('/');
      await expect(page.locator('[data-gate-status="complete"]')).toBeVisible();
    }

    test.describe('Phase 2 shell', () => {
      test('SC-3: ⌘K palette opens, fuzzy filter, Enter navigates, Esc closes', async ({ page, browserName }) => {
        await bootAsReturning(page);

        const modKey = browserName === 'webkit' ? 'Meta' : 'Control';
        await page.keyboard.press(`${modKey}+KeyK`);
        await expect(page.getByRole('dialog', { name: /command palette/i })).toBeVisible();

        const search = page.getByRole('textbox', { name: /search routes/i });
        await search.fill('settings');
        await expect(page.locator('[role="option"]').first()).toContainText(/settings/i);

        await page.keyboard.press('Enter');
        await expect(page.getByRole('dialog', { name: /command palette/i })).toHaveCount(0);
        await expect(page.locator('[data-route-id="settings"]')).toBeVisible();

        // Re-open and close with Escape
        await page.keyboard.press(`${modKey}+KeyK`);
        await expect(page.getByRole('dialog', { name: /command palette/i })).toBeVisible();
        await page.keyboard.press('Escape');
        await expect(page.getByRole('dialog', { name: /command palette/i })).toHaveCount(0);
      });

      test('SC-3b: adding a route via index.tsx surfaces in palette (regression proof)', async ({ page, browserName }) => {
        // This test just asserts that PALETTE_COMMANDS is read live — the actual
        // "add a route" surface test is a development-time check, not runtime.
        // We falsify the live-read assumption by confirming every existing
        // palette-visible route appears when query is empty.
        await bootAsReturning(page);
        const modKey = browserName === 'webkit' ? 'Meta' : 'Control';
        await page.keyboard.press(`${modKey}+KeyK`);
        await expect(page.getByRole('dialog', { name: /command palette/i })).toBeVisible();
        // Empty query should surface at least dashboard, chat, and settings (3 core).
        await expect(page.locator('[role="option"][data-route-id="dashboard"]')).toHaveCount(1);
        await expect(page.locator('[role="option"][data-route-id="chat"]')).toHaveCount(1);
        await expect(page.locator('[role="option"][data-route-id="settings"]')).toHaveCount(1);
        // Onboarding is paletteHidden — should NOT appear.
        await expect(page.locator('[role="option"][data-route-id="onboarding"]')).toHaveCount(0);
        await page.keyboard.press('Escape');
      });

      test('SC-4: backend blade_notification → toast renders and auto-dismisses', async ({ page }) => {
        await bootAsReturning(page);

        // Dispatch the event via the shim's CustomEvent bus (BackendToastBridge
        // subscribes to BLADE_EVENTS.BLADE_NOTIFICATION = 'blade_notification').
        await page.evaluate(() => {
          window.dispatchEvent(new CustomEvent('tauri://blade_notification', {
            detail: { type: 'info', message: 'Hello from Rust' },
          }));
        });

        const toast = page.locator('.toast[data-toast-type="info"]');
        await expect(toast).toBeVisible();
        await expect(toast).toContainText(/hello from rust/i);

        // Auto-dismiss: default info duration is 4000ms. Wait slightly longer.
        await expect(toast).toHaveCount(0, { timeout: 6_000 });

        // Interaction not blocked: palette still opens after the toast cycle.
        await page.keyboard.press('Control+KeyK');
        await expect(page.getByRole('dialog', { name: /command palette/i })).toBeVisible();
        await page.keyboard.press('Escape');
      });
    });
