    #!/usr/bin/env node
    // Parses vite.config.ts rollupOptions.input, asserts every declared HTML exists on disk.
    import { readFileSync, existsSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const ROOT = resolve(fileURLToPath(import.meta.url), '..', '..');
    const cfg = readFileSync(resolve(ROOT, 'vite.config.ts'), 'utf8');
    const inputs = [...cfg.matchAll(/(\w+):\s*resolve\(__dirname,\s*"([^"]+)"\)/g)]
      .map(m => ({ key: m[1], path: m[2] }));

    if (inputs.length === 0) {
      console.error('[verify-entries] FAIL: could not parse any inputs from vite.config.ts');
      process.exit(1);
    }

    let failed = false;
    for (const { key, path } of inputs) {
      const full = resolve(ROOT, path);
      if (!existsSync(full)) {
        console.error(`[verify-entries] MISSING: ${key} -> ${path}`);
        failed = true;
      }
    }
    if (failed) process.exit(1);
    console.log(`[verify-entries] OK — ${inputs.length} entries present`);

    ### 1d. `/home/arnav/blade/scripts/verify-migration-ledger.mjs` (D-27, P-03)
    ```javascript
    #!/usr/bin/env node
    // Parses .planning/migration-ledger.md, greps src/ for route-id references,
    // fails if any src/ reference targets a route whose ledger status is not Pending/Shipped.
    import { readFileSync, readdirSync, statSync } from 'node:fs';
    import { join, resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const ROOT = resolve(fileURLToPath(import.meta.url), '..', '..');
    const ledger = readFileSync(join(ROOT, '.planning', 'migration-ledger.md'), 'utf8');
    const rows = {};
    for (const line of ledger.split('\n')) {
      const m = line.match(/^\|\s*([a-z-]+)\s*\|[^|]+\|[^|]+\|[^|]+\|[^|]+\|\s*(Pending|Shipped|Deferred)\s*\|/);
      if (m) rows[m[1]] = m[2];
    }
    console.log(`[verify-migration-ledger] ${Object.keys(rows).length} rows loaded`);

    // Walk src/ for openRoute('<id>') / routeId === '<id>' references
    function* walk(dir) {
      for (const e of readdirSync(dir)) {
        const p = join(dir, e);
        if (statSync(p).isDirectory()) yield* walk(p); else if (/\.(tsx?|jsx?)$/.test(e)) yield p;
      }
    }

    const referencedIds = new Set();
    for (const f of walk(join(ROOT, 'src'))) {
      const src = readFileSync(f, 'utf8');
      for (const m of src.matchAll(/openRoute\(\s*['"]([a-z-]+)['"]/g)) referencedIds.add(m[1]);
      for (const m of src.matchAll(/routeId\s*===?\s*['"]([a-z-]+)['"]/g)) referencedIds.add(m[1]);
    }

    let failed = false;
    for (const id of referencedIds) {
      if (!(id in rows)) {
        console.error(`[verify-migration-ledger] ORPHAN: src/ references '${id}' but ledger has no row`);
        failed = true;
      }
    }
    if (failed) process.exit(1);
    console.log(`[verify-migration-ledger] OK — ${referencedIds.size} referenced ids all tracked`);
    ```

    ### 1f. `/home/arnav/blade/scripts/audit-contrast.mjs` (D-33, P-08)
    ```javascript
    #!/usr/bin/env node
    // WCAG 2.1 contrast audit of documented text-on-glass pairs from tokens.css.
    // Composites glass rgba over a realistic dark wallpaper baseline, computes relative luminance,
    // fails if any pair < 4.5:1.
    import { readFileSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const ROOT = resolve(fileURLToPath(import.meta.url), '..', '..');
    const tokens = readFileSync(resolve(ROOT, 'src/styles/tokens.css'), 'utf8');

    // Extract rgba values by token name
    const rgba = {};
    for (const m of tokens.matchAll(/--([a-z][\w-]+):\s*rgba\(([\d.,\s]+)\)/gi)) {
      rgba[m[1]] = m[2].split(',').map(n => parseFloat(n.trim()));
    }

    // Realistic dark wallpaper baseline (shared.css deep indigo #0a0a1d ish)
    const BG_WALLPAPER = [10, 10, 29];

    function composite([r1, g1, b1, a1], [r2, g2, b2]) {
      return [
        Math.round(r1 * a1 + r2 * (1 - a1)),
        Math.round(g1 * a1 + g2 * (1 - a1)),
        Math.round(b1 * a1 + b2 * (1 - a1)),
      ];
    }

    function relLuminance([r, g, b]) {
      const toLin = c => {
        c = c / 255;
        return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
      };
      return 0.2126 * toLin(r) + 0.7152 * toLin(g) + 0.0722 * toLin(b);
    }

    function contrast(fg, bg) {
      const L1 = relLuminance(fg);
      const L2 = relLuminance(bg);
      return (Math.max(L1, L2) + 0.05) / (Math.min(L1, L2) + 0.05);
    }

    // Text (composited over composited glass over wallpaper)
    const pairs = [
      { label: 't-1 on glass-1', fgKey: 't-1', bgKey: 'g-fill' },
      { label: 't-2 on glass-1', fgKey: 't-2', bgKey: 'g-fill' },
      { label: 't-1 on glass-2', fgKey: 't-1', bgKey: 'g-fill-strong' },
      { label: 't-2 on glass-2', fgKey: 't-2', bgKey: 'g-fill-strong' },
      { label: 't-1 on glass-3', fgKey: 't-1', bgKey: 'g-fill-heavy' },
      { label: 't-2 on glass-3', fgKey: 't-2', bgKey: 'g-fill-heavy' },
      { label: 't-3 on glass-1', fgKey: 't-3', bgKey: 'g-fill' },   // borderline — may require design nudge
    ];

    let failed = false;
    for (const { label, fgKey, bgKey } of pairs) {
      const glassOver = composite(rgba[bgKey], BG_WALLPAPER);
      const fgOver = composite(rgba[fgKey], glassOver);
      const ratio = contrast(fgOver, glassOver);
      const mark = ratio >= 4.5 ? 'PASS' : 'FAIL';
      console.log(`[audit-contrast] ${mark} ${label}: ${ratio.toFixed(2)}:1`);
      if (ratio < 4.5) failed = true;
    }
    if (failed) {
      console.error('[audit-contrast] FAIL: one or more text/bg pairs fail WCAG AA 4.5:1');
      process.exit(1);
    }
    console.log('[audit-contrast] OK — all pairs ≥ 4.5:1 on dark wallpaper baseline');
    ```

    // ESLint flat-config rule: ban raw @tauri-apps/api/core invoke + @tauri-apps/api/event listen
    // outside allowed paths.
    export default {
      meta: {
        type: 'problem',
        docs: { description: 'Ban raw Tauri invoke/listen outside src/lib/tauri and src/lib/events' },
        messages: {
          rawInvoke: 'Use invokeTyped from @/lib/tauri/_base instead of raw invoke (D-34)',
          rawListen: 'Use useTauriEvent from @/lib/events instead of raw listen (D-34)',
        },
        schema: [],

        const isAllowedInvoke = filename.includes('/src/lib/tauri/');
        const isAllowedListen = filename.includes('/src/lib/events/');
        return {
          ImportDeclaration(node) {
            const src = node.source.value;
            if (src === '@tauri-apps/api/core' && !isAllowedInvoke) {
              const hasInvoke = node.specifiers.some(
                s => s.type === 'ImportSpecifier' && s.imported && s.imported.name === 'invoke'

            if (src === '@tauri-apps/api/event' && !isAllowedListen) {

                s => s.type === 'ImportSpecifier' && s.imported && s.imported.name === 'listen'

        };
      },

    // src/features/dev/WrapperSmoke.tsx — DEV-only, palette-hidden.
    // P-04 gate verification: exercises every Phase-1 typed wrapper and reports pass/fail.
    import { useCallback, useState } from 'react';
    import { Card, Button, Pill } from '@/design-system/primitives';
    import {
      getConfig, getOnboardingStatus, sendMessageStream, cancelChat, TauriError,
    } from '@/lib/tauri';

    interface TestRow {
      name: string;
      rustCite: string;
      argsPreview: string;
      status: 'idle' | 'running' | 'pass' | 'fail';
      result?: string;
      error?: string;
    }

    const TESTS: Omit<TestRow, 'status'>[] = [
      { name: 'getConfig', rustCite: 'commands.rs:1899', argsPreview: '{}' },
      { name: 'getOnboardingStatus', rustCite: 'commands.rs:2312', argsPreview: '{}' },
      { name: 'cancelChat (idempotent)', rustCite: 'commands.rs:71', argsPreview: '{}' },
      // sendMessageStream / saveConfig / completeOnboarding are NOT auto-run — they mutate state.
      // The table marks them as 'idle' with a note; operator can trigger manually.
    ];

    export function WrapperSmoke() {
      const [rows, setRows] = useState<TestRow[]>(() =>
        TESTS.map(t => ({ ...t, status: 'idle' }))
      );

      const runAll = useCallback(async () => {
        setRows(prev => prev.map(r => ({ ...r, status: 'running', result: undefined, error: undefined })));
        const updated: TestRow[] = [];
        for (const t of TESTS) {
          const row: TestRow = { ...t, status: 'running' };
          try {
            let result: unknown;
            if (t.name === 'getConfig') result = await getConfig();
            else if (t.name === 'getOnboardingStatus') result = await getOnboardingStatus();
            else if (t.name === 'cancelChat (idempotent)') result = await cancelChat();
            row.status = 'pass';
            row.result = typeof result === 'object' ? JSON.stringify(result).slice(0, 80) : String(result);
          } catch (e) {
            row.status = 'fail';
            row.error = e instanceof TauriError ? `[${e.kind}] ${e.rustMessage}` : String(e);
          }
          updated.push(row);
        }
        setRows(updated);
      }, []);

      return (
        <div style={{ padding: 'var(--s-8)' }}>
          <h1 className="t-h1">Wrapper Smoke (DEV)</h1>
          <p className="t-body" style={{ color: 'var(--t-2)', margin: 'var(--s-3) 0' }}>

                <tr style={{ textAlign: 'left', color: 'var(--t-3)', fontFamily: 'var(--font-mono)' }}>
                  <th style={{ padding: 'var(--s-2)' }}>Fn</th>
                  <th style={{ padding: 'var(--s-2)' }}>Rust cite</th>

                  <th style={{ padding: 'var(--s-2)' }}>Status</th>
                  <th style={{ padding: 'var(--s-2)' }}>Result / Error</th>
                </tr>
              </thead>
              <tbody>

                  <tr key={r.name} style={{ borderTop: '1px solid var(--line)' }}>
