<!-- Seed script pattern from PATTERNS.md §scripts/seed-migration-ledger.mjs -->

`scripts/seed-migration-ledger.mjs`:
```javascript
#!/usr/bin/env node
// Walks src/features/*/index.tsx to extract RouteDefinition rows,
// matches each to a src.bak/components/*.tsx analog (filename heuristic),
// writes .planning/migration-ledger.md with the full table.
// Idempotent: re-running preserves existing `status` column values for rows that still exist.

import { readdirSync, readFileSync, writeFileSync, existsSync, statSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const ROOT = join(fileURLToPath(import.meta.url), '..', '..');

// 1. Walk feature indexes, parse RouteDefinition entries
function parseFeatureIndex(path) {
  const src = readFileSync(path, 'utf8');
  const matches = [...src.matchAll(
    /\{\s*id:\s*'([a-z-]+)',\s*label:\s*'([^']+)',\s*section:\s*'([a-z]+)'[\s\S]*?phase:\s*(\d+)/g
  )];
  return matches.map(m => ({ id: m[1], label: m[2], section: m[3], phase: +m[4] }));
}

// 2. Find src.bak analog (filename heuristic)
function findBakAnalog(routeId, label) {
  const bakDir = join(ROOT, 'src.bak', 'components');
  if (!existsSync(bakDir)) return 'N/A';
  const files = readdirSync(bakDir);
  const pascal = label.replace(/\s+/g, '').replace(/[^A-Za-z0-9]/g, '');
  // Exact filename match, then fuzzy
  const exact = files.find(f => f.toLowerCase() === `${pascal.toLowerCase()}.tsx`);
  if (exact) return `src.bak/components/${exact}`;
  const fuzzy = files.find(f => f.toLowerCase().includes(pascal.toLowerCase().slice(0, 6)));
  return fuzzy ? `src.bak/components/${fuzzy}` : 'N/A (new)';
}

// 3. Preserve existing status column if re-seeding
function loadExistingStatus(ledgerPath) {
  if (!existsSync(ledgerPath)) return {};
  const text = readFileSync(ledgerPath, 'utf8');
  const statusMap = {};
  for (const line of text.split('\n')) {
    const m = line.match(/^\|\s*([a-z-]+)\s*\|[^|]+\|[^|]+\|[^|]+\|[^|]+\|\s*(Pending|Shipped|Deferred)\s*\|/);
    if (m) statusMap[m[1]] = m[2];
  }
  return statusMap;
}

// 4. Collect all routes from all feature files
const featuresDir = join(ROOT, 'src', 'features');
const clusters = readdirSync(featuresDir).filter(d => statSync(join(featuresDir, d)).isDirectory());
const allRoutes = [];
for (const c of clusters) {
  const indexFile = join(featuresDir, c, 'index.tsx');
  if (!existsSync(indexFile)) continue;
  for (const r of parseFeatureIndex(indexFile)) {
    allRoutes.push({ ...r, cluster: c });
  }
}

// 5. Build ledger markdown
const ledgerPath = join(ROOT, '.planning', 'migration-ledger.md');
const existingStatus = loadExistingStatus(ledgerPath);

const rows = allRoutes.map(r => {
  const bak = findBakAnalog(r.id, r.label);
  const newComp = `src/features/${r.cluster}/${r.label.replace(/\s+/g, '')}.tsx`;
  const status = existingStatus[r.id] ?? 'Pending';
  return `| ${r.id} | ${bak} | ${newComp} | ${r.section} | ${r.phase} | ${status} | – |  |`;
});

const out = `# BLADE Migration Ledger

**Seeded:** ${new Date().toISOString().slice(0, 10)} (Plan 08 auto-generated)
**Discipline:** D-27 + D-28 + P-03.
**Enforcement:** CI via \`scripts/verify-migration-ledger.mjs\` (Plan 09).

## Invariants
1. No old route removed before its new component ships (status \`Pending\` → \`Shipped\`).
2. \`cross_refs\` column MUST be cleared before flipping status to \`Shipped\`.
3. Adding a new route appends a row (FOUND-08 acceptance: 1 file + 1 entry).
4. Re-run \`npm run seed:ledger\` after adding routes to refresh the table without losing status.

## Rows

| route_id | src.bak_path | new_component | section | phase | status | cross_refs | notes |
|----------|--------------|---------------|---------|-------|--------|------------|-------|
${rows.join('\n')}

## Totals

- Routes tracked: ${rows.length}
- Status breakdown: Pending=${rows.filter(r => r.includes('| Pending |')).length}, Shipped=${rows.filter(r => r.includes('| Shipped |')).length}
`;

writeFileSync(ledgerPath, out, 'utf8');
console.log(`[seed-migration-ledger] wrote ${rows.length} rows to ${ledgerPath}`);
```
