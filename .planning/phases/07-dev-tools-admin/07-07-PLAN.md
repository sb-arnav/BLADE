    ```tsx
    // src/features/dev/TerminalDev.tsx
    // Dev-only isolation harness for Playwright dev-tools-terminal.spec.ts.
    import { useEffect, useState } from 'react';
    import { Terminal } from '@/features/dev-tools/Terminal';

    const MOCK_SHELL_RESULT = { stdout: 'mock-output\nmock-line-2\n', stderr: '', exit_code: 0 };

    export function TerminalDev() {
      const [ready, setReady] = useState(false);
      useEffect(() => {
        const prev = window.__TAURI_INVOKE_HOOK__;
        const hook = new Map<string, (args: unknown) => unknown>();
        hook.set('run_shell', () => MOCK_SHELL_RESULT);
        hook.set('run_code_block', () => MOCK_SHELL_RESULT);
        hook.set('ask_ai', () => 'Mock AI answer');
        window.__TAURI_INVOKE_HOOK__ = hook;
        setReady(true);
        return () => { window.__TAURI_INVOKE_HOOK__ = prev; };
      }, []);
      if (!ready) return null;
      return <Terminal />;
    }
    ```
