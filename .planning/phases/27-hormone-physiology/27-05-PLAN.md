**Test entry point:**
```rust
#[test]
fn evaluates_hormone_physiology() {
    let cases = fixtures();
    assert!(cases.len() >= 9, "Expected at least 9 fixtures, got {}", cases.len());
    let mut rows: Vec<EvalRow> = Vec::with_capacity(cases.len());
    for fx in &cases {
        let (passed, detail) = (fx.run)();
        rows.push(EvalRow {
            label: fx.label.to_string(),
            passed,
            score: if passed { 1.0 } else { 0.0 },
            detail,
        });
    }
    harness::print_eval_table("Hormone physiology eval", &rows);
    let s = harness::summarize(&rows);
    let floor_passed = s.pass_rate >= MODULE_FLOOR;
    harness::record_eval_run(MODULE_NAME, &s, floor_passed);
    assert!(floor_passed, "Hormone eval below {:.0}% floor: {:.1}%", MODULE_FLOOR * 100.0, s.pass_rate * 100.0);
}
```

Use a `HormoneFixture` struct:
```rust
struct HormoneFixture {
    label: &'static str,
    run: fn() -> (bool, String),
}
```

The `fixtures()` function returns `Vec<HormoneFixture>` with the 9 fixtures above. Each fixture's `run` function returns `(passed: bool, detail: String)`.
