/// Pattern 9 (need_integration) returns CapabilityGap per D-13.
/// Patterns 1-8 (refusals) return Refusal unless the disjunction post-check fires
/// (next 80 chars contain `\bbut\b.+\bcan\b`), in which case return Pass.
pub fn intercept_assistant_output(transcript: &str) -> EgoVerdict {
    static DISJUNCTION_POSTCHECK: OnceLock<Regex> = OnceLock::new();
    let post = DISJUNCTION_POSTCHECK.get_or_init(|| Regex::new(r"\bbut\b.+\bcan\b").unwrap());

    for (re, label) in refusal_patterns().iter() {
        if let Some(m) = re.find(transcript) {
            // Pattern 9 → CapabilityGap (D-13 precedence)
            if *label == "need_integration" {
                // Extract the capability noun (the \w+ between "I'd need" and "integration|tool|api")
                let captured = re.captures(transcript)
                    .and_then(|c| c.get(0))
                    .map(|m| m.as_str().to_string())
                    .unwrap_or_default();
                let capability = captured
                    .split_whitespace()
                    .nth(2)  // "I'd need a CAPABILITY integration"
                    .unwrap_or("unknown")
                    .to_string();
                return EgoVerdict::CapabilityGap {
                    capability,
                    suggestion: "Connect via Integrations tab".to_string(),
                };
            }
            // Patterns 1-8 → Refusal (with disjunction post-check)
            // Slice the next 80 chars from the match end (safe_slice for non-ASCII).
            let end = m.end();
            let lookahead_end = transcript.len().min(end + 80);
            // Use char-boundary-safe substring via safe_slice or careful indexing.
            let lookahead = if let Some(slice) = transcript.get(end..lookahead_end) {
                slice
            } else {
                // Fall back to safe_slice if the byte range crosses a UTF-8 boundary.
                let remaining = &transcript[end..];
                let safe = crate::safe_slice(remaining, 80);
                // safe_slice returns owned String; we need &str — accept the false-positive
                // risk for the rare cross-boundary case (unit-test the common ASCII path).
                if post.is_match(&safe) {
                    return EgoVerdict::Pass;
                }
                ""
            };
            if post.is_match(lookahead) {
                return EgoVerdict::Pass;
            }
            return EgoVerdict::Refusal {
                pattern: label.to_string(),
                reason: m.as_str().to_string(),
            };
        }
    }
    EgoVerdict::Pass
}

/// Phase 18 — handle_refusal per CONTEXT D-14 (retry cap = 1) / D-15 (locked output format) / D-16 (Runtime vs Integration).
pub async fn handle_refusal(app: &AppHandle, verdict: EgoVerdict, original: &str) -> EgoOutcome {
    // D-14: retry cap = 1 per turn. fetch_add returns the PREVIOUS value, so >= 1 means we already used the retry.
    let prev = RETRY_COUNT.fetch_add(1, Ordering::SeqCst);
    if prev >= 1 {
        // Already retried once this turn; persistent refusal → hard_refuse without further work.
        let final_response = format!(
            "I tried, but I exhausted my retry budget for this turn. Try rephrasing or starting a new turn."
        );
        emit_jarvis_intercept(app, "hard_refused", None, Some("retry_cap_exceeded"));
        return EgoOutcome::HardRefused {
            final_response,
            logged_gap: false,
        };
    }

    match verdict {
        EgoVerdict::Pass => {
            // Defensive — handle_refusal shouldn't be called for Pass.
            EgoOutcome::Retried { new_response: original.to_string() }
        }
        EgoVerdict::CapabilityGap { capability, suggestion: _ } => {
            // 1. Log the gap (reuses evolution.rs:1115 verbatim).
            emit_jarvis_intercept(app, "intercepting", Some(&capability), None);
            let _ = crate::evolution::evolution_log_capability_gap(
                capability.clone(),
                crate::safe_slice(original, 200),
            );

            // 2. Look up the catalog entry; route by kind (D-16).
            let catalog = crate::self_upgrade::capability_catalog();
            let key = capability.to_lowercase();
            // Try several lookups to find the catalog entry: exact, with _outbound suffix, etc.
            let entry = catalog.get(key.as_str())
                .or_else(|| catalog.get(format!("{}_outbound", key).as_str()))
                .or_else(|| catalog.get(format!("{}_write", key).as_str()))
                .cloned();

            match entry {
                Some(gap) if matches!(gap.kind, crate::self_upgrade::CapabilityKind::Integration) => {
                    // Integration kind: D-15 locked hard-refuse format with integration_path.
                    let final_response = format!(
                        "I tried, but I don't have a {} integration. Here's what I'd need: {}. You can connect it via {}.",
                        capability,
                        gap.description,
                        gap.integration_path
                    );
                    emit_jarvis_intercept(app, "hard_refused", Some(&capability), Some(&gap.description));
                    EgoOutcome::HardRefused {
                        final_response,
                        logged_gap: true,
                    }
                }
                Some(gap_runtime) => {
                    // Runtime kind: emit installing → call auto_install with the catalog entry → emit retrying.
                    // LIVE auto_install signature (verified W2 pre-pin): `async fn auto_install(gap: &CapabilityGap) -> InstallResult`.
                    // InstallResult { tool: String, success: bool, output: String }.
                    emit_jarvis_intercept(app, "installing", Some(&capability), None);
                    let install_result = crate::self_upgrade::auto_install(&gap_runtime).await;
                    if !install_result.success {
                        // Install failed — fall through to HardRefused with the install output as reason.
                        let final_response = format!(
                            "I tried, but I couldn't install {}. Here's what I'd need: {}. You can connect it manually via Integrations tab.",
                            capability, crate::safe_slice(&install_result.output, 100)
                        );
                        emit_jarvis_intercept(app, "hard_refused", Some(&capability), Some("install_failed"));
                        return EgoOutcome::HardRefused { final_response, logged_gap: true };
                    }
                    emit_jarvis_intercept(app, "retrying", Some(&capability), None);
                    EgoOutcome::AutoInstalled {
                        capability: capability.clone(),
                        // then_retried filled in by commands.rs (Plan 10) wrapper which handles the retry call to the LLM.
                        then_retried: format!("<retry-pending: {} installed via {}>", capability, install_result.tool),
                    }
                }
                None => {
                    // No catalog entry — D-15 hard-refuse with generic reason.
                    let final_response = format!(
                        "I tried, but I don't have a {} capability. Here's what I'd need: {} support. You can connect it via Integrations tab.",
                        capability, capability
                    );
                    emit_jarvis_intercept(app, "hard_refused", Some(&capability), Some("no_catalog_entry"));
                    EgoOutcome::HardRefused {
                        final_response,
                        logged_gap: true,
                    }
                }
            }
        }
        EgoVerdict::Refusal { pattern, reason } => {
            // Bare refusal (no capability_gap precursor) — log as gap, hard-refuse.
            emit_jarvis_intercept(app, "intercepting", None, Some(&pattern));
            let _ = crate::evolution::evolution_log_capability_gap(
