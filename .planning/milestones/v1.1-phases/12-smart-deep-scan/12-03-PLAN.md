```rust
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
#[serde(rename_all = "snake_case")]
pub enum OverlayAction { Edit, Hide, Delete, Add }

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct OverlayEntry {
    pub action: OverlayAction,
    pub fields: Option<std::collections::HashMap<String, serde_json::Value>>,
    pub edited_at: chrono::DateTime<chrono::Utc>,
    pub not_found: bool,   // true when row_id absent from latest scan
}

#[derive(Debug, Clone, Serialize, Deserialize, Default)]
pub struct ProfileOverlay {
    pub version: u32,     // always 1
    pub rows: std::collections::HashMap<String, OverlayEntry>,
}

    pub edited: bool,

}
```

    ```rust
    fn overlay_path() -> PathBuf {
        crate::config::blade_config_dir().join("identity").join("profile_overlay.json")

    pub fn save_overlay(overlay: &ProfileOverlay) -> Result<(), String> {
        let path = overlay_path();
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent).map_err(|e| e.to_string())?;
        }
        let json = serde_json::to_string_pretty(overlay).map_err(|e| e.to_string())?;
        // Atomic write: write to temp file then rename
        let tmp = path.with_extension("json.tmp");
        std::fs::write(&tmp, &json).map_err(|e| e.to_string())?;
        std::fs::rename(&tmp, &path).map_err(|e| e.to_string())?;
        Ok(())
    }

    pub fn load_overlay() -> ProfileOverlay {
        let path = overlay_path();
        match std::fs::read_to_string(&path) {
            Ok(s) => serde_json::from_str(&s).unwrap_or_default(),
            Err(_) => ProfileOverlay::default(),
        }
    }
    ```

    Implement the D-62 render-layer merge algorithm verbatim:
    ```
    for each row in scan results (repos, accounts, mru_files, tools, ides, bookmarks):
        overlay_entry = overlay.rows.get(&row.row_id)
        if action == "delete" → skip
        if action == "hide"   → skip
        if action == "edit"   → render row with overlay.fields applied (overlay wins per-field), set edited=true
        else                  → render row as-is, edited=false
    for each (row_id, entry) in overlay where action == "add":
        render as RenderedRow with source_scanner="user", orphaned=false
    for each (row_id, entry) in overlay where row_id NOT in scan AND action != "add" AND action != "delete":
        render as orphaned RenderedRow (orphaned=true, not_found pill)
    ```
    When applying overlay fields to a scan row: convert the scan row to `serde_json::Value` (via serde_json::to_value), then merge overlay.fields on top (overlay value wins for any key present in both). Convert back to `RenderedRow.fields: HashMap<String, Value>`.

    ```rust
    #[tauri::command]
    pub async fn profile_get_rendered() -> Result<ProfileView, String> {
        let scan = crate::deep_scan::mod::load_results();  // adjust path as needed
        let overlay = load_overlay();
        let view = match scan {
            Some(s) => merge_scan_with_overlay(&s, &overlay),
            None => ProfileView::default(),
        };
        Ok(view)
    }

    #[tauri::command]
    pub async fn profile_overlay_upsert(
        row_id: String,
        action: OverlayAction,
        fields: Option<std::collections::HashMap<String, serde_json::Value>>,
    ) -> Result<(), String> {
        let _guard = OVERLAY_LOCK.lock().await;
        let mut overlay = load_overlay();
        overlay.rows.insert(row_id, OverlayEntry {
            action,
            fields,
            edited_at: chrono::Utc::now(),
            not_found: false,
        });
        save_overlay(&overlay)
    }

    #[tauri::command]
    pub async fn profile_overlay_reset(row_id: String) -> Result<(), String> {
        let _guard = OVERLAY_LOCK.lock().await;
        let mut overlay = load_overlay();
        overlay.rows.remove(&row_id);
        save_overlay(&overlay)
    }

    #[tauri::command]
    pub async fn scan_cancel() -> Result<(), String> {
        crate::deep_scan::queue::SCAN_CANCEL.store(true, std::sync::atomic::Ordering::SeqCst);
        Ok(())
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Frontend → profile_overlay_upsert | Row edits from the renderer (row_id + action + fields) — row_id is user-constructed; fields are user-supplied values |
| profile_overlay.json on disk | File is user-readable; contains user edits but no new secrets (row_ids are paths/handles already in scan_results.json) |
| Concurrent UI save during active scan | scan thread reads overlay; UI writes overlay; OVERLAY_LOCK serializes writes |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-13 | Tampering | profile_overlay_upsert row_id injection | accept | row_id is used as a HashMap key; no filesystem path constructed from it in the overlay writer. Keys are written to a JSON file, not executed. Low risk. |
| T-12-14 | Denial of Service | profile_overlay.json corrupt/huge | mitigate | `serde_json::from_str` failure → `unwrap_or_default()` returns empty overlay; scan rows render normally without edits. User edits are lost on corruption, not the scan data. |
| T-12-15 | Tampering | Concurrent overlay writes (UI save during scan) | mitigate | `static OVERLAY_LOCK: tokio::sync::Mutex<()>` serializes all overlay writes; atomic rename prevents partial-write reads. |
| T-12-16 | Information Disclosure | profile_overlay.json readable by other processes | accept | File contains user-chosen edits only (no new secrets beyond what scan_results.json already stored). Permissions inherit directory umask (same as scan_results.json). |
| T-12-17 | Spoofing | scan_cancel called by malicious renderer | accept | Cancel only stops the local scan; no data is lost; scan can be restarted. Low-risk operation for a local-only app. |
</threat_model>
