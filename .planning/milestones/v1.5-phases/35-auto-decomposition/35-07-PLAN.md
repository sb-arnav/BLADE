/// Plan 35-07 — REAL sub-agent dispatch. Constructs sub-agent's
/// LoopState{is_subagent=true} + SessionWriter pointing at forked JSONL +
/// invokes run_loop with the StepGroup goal as the synthetic User message.
/// The recursion gate in run_loop's pre-iteration trigger (Plan 35-04)
/// prevents grandchildren.
///
/// Returns (success, cost_usd, tokens_used) where:
///   - success: true if halt was IterationCap or clean exit; false on
///     CostExceeded / Stuck / CircuitOpen / ProviderFatal / Cancelled
///   - cost_usd: sub-agent's per-loop cumulative_cost_usd at halt
///   - tokens_used: estimated from sub-agent's recent_actions / state
async fn run_subagent_to_halt(
    new_id: &str,
    group: &StepGroup,
    app: &AppHandle,
    config: &BladeConfig,
) -> Result<(bool, f32, u32), String> {
    // (a) Construct sub-agent state — recursion gate ON.
    let mut state = LoopState::default();
    state.is_subagent = true;

    // (b) Seed conversation with StepGroup goal as synthetic User message.
    //     The sub-agent's own LLM interprets the goal; the role's system
    //     prompt snippet (AgentRole::system_prompt_snippet) is prepended
    //     by the run_loop's brain prompt assembly.
    let mut conversation: Vec<crate::providers::ConversationMessage> = vec![
        crate::providers::ConversationMessage::User(group.goal.clone()),
    ];

    // (c) Provider selection — use existing select_provider_for_task with
    //     the role hint. Falls back to parent's provider if role-preferred
    //     has no API key.
    let _role = group.role.clone();
    // The existing swarm_commands::select_provider_for_task takes a SwarmTask;
    // for Phase 35 we either expose a public role-only variant or replicate
    // the logic inline. Recommend exposing crate::swarm_commands::select_provider_for_role
    // as a thin shim. For Plan 35-07 v1, fall through to parent provider —
    // role-aware routing is a v1.6 polish.
    let _provider = config.provider.clone();
    let _api_key = config.api_key.clone();
    let _model = config.model.clone();

    // (d) Open SessionWriter on the forked JSONL.
    //     SessionWriter::new_for_id constructs a writer pointing at an
    //     existing session_id (the forked ULID from Plan 35-05's
    //     fork_session call). If the helper doesn't exist, add it to
    //     session/log.rs as a thin wrapper:

    //           pub fn open_existing(dir: &Path, id: &str, enabled: bool) -> Self {
    //               Self {
    //                   path: dir.join(format!("{}.jsonl", id)),
    //                   enabled,
    //               }
    //           }
    //       }
    let _writer = crate::session::log::SessionWriter::open_existing(
        &config.session.jsonl_log_dir, new_id, config.session.jsonl_log_enabled
    );

    // (e) Invoke run_loop.
    //     EXACT call signature depends on the current run_loop fn shape —
    //     adapt as needed. Plan 35-07 EXECUTES with whatever is current at
    //     master. The minimal contract:
    //       - run_loop(&mut state, &mut conversation, &mut writer, ...)

    //     execute_decomposed_task's tests actually exercise.
