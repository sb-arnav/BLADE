// ── EVAL-02: Context efficiency fixtures ──────────────────────────────────
//
// Plan 37-04 / EVAL-02 — 3 fixtures asserting LAST_BREAKDOWN section presence
// + total-token cap. CONTEXT lock §EVAL-02. No live LLM calls — pure prompt
// assembly inspection via brain::build_system_prompt + read_section_breakdown.

#[cfg(test)]
struct ContextEfficiencyFixture {
    label: &'static str,
    query: &'static str,
    expected_max_total_tokens: usize,
    forbidden_section_labels: &'static [&'static str],
    required_section_labels: &'static [&'static str],
}

#[cfg(test)]
fn eval_02_total_tokens(breakdown: &[(String, usize)]) -> usize {
    // Token estimation reuses Phase 32's chars/4 helper (CONTEXT lock §EVAL-02).
    breakdown.iter().map(|(_, chars)| *chars).sum::<usize>() / 4
}

#[cfg(test)]
fn eval_02_run(fix: &ContextEfficiencyFixture) -> (bool, String) {
    // Clear the breakdown accumulator before each run (Phase 32-CTX-06 contract:
    // build_system_prompt_inner clears+repopulates on each invocation; explicit
    // clear here ensures we don't read leftovers from a prior eval).
    crate::brain::clear_section_breakdown();

    // Call the Tauri-friendly wrapper. The wrapper internally calls
    // build_system_prompt_inner with default tier + provider + model + zero
    // anchor injections — sufficient for context-efficiency assertions.
    let _prompt = crate::brain::build_system_prompt(&[], fix.query, None);

    let breakdown = crate::brain::read_section_breakdown();
    let total_tokens = eval_02_total_tokens(&breakdown);
    let labels: Vec<String> = breakdown.iter().map(|(l, _)| l.clone()).collect();

    let cap_ok = total_tokens <= fix.expected_max_total_tokens;
    let no_forbidden = !fix.forbidden_section_labels.iter()
        .any(|forbid| labels.iter().any(|l| l == forbid));
    let all_required = fix.required_section_labels.iter()
        .all(|req| labels.iter().any(|l| l == req));

    let cfg = crate::config::load_config();
    let strict = cfg.eval.context_efficiency_strict;
    let passed = if strict {
        cap_ok && no_forbidden && all_required
    } else {
        // Soft-warn mode: only required labels are asserted; cap and forbidden
        // produce warnings but do not fail the row. CONTEXT lock §EvalConfig
        // Sub-Struct §Locked: context_efficiency_strict.
        all_required
    };

    let summary = format!(
        "{}: total={}t (cap {}t, ok={}), no_forbidden={}, all_required={}, labels=[{}]",
        fix.label, total_tokens, fix.expected_max_total_tokens, cap_ok,
        no_forbidden, all_required, labels.join(", ")
    );
    (passed, summary)
}
