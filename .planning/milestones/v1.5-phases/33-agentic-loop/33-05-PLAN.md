```rust
let (content, is_error) = match dispatch_result {
    Ok(s) => {
        if config.r#loop.smart_loop_enabled {
            state.consecutive_same_tool_failures.clear();
        }
        (s, false)
    }
    Err(e) => {
        let mut tool_err = crate::native_tools::wrap_legacy_error(&call.name, e);
        if config.r#loop.smart_loop_enabled {
            tool_err.suggested_alternatives = enrich_alternatives(&call.name);

            // Reset on tool change, then increment.
            let last_failed_tool = state.consecutive_same_tool_failures.keys().next().cloned();
            if let Some(prev) = last_failed_tool {
                if prev != call.name {
                    state.consecutive_same_tool_failures.clear();
                }
            }
            let counter = state.consecutive_same_tool_failures
                .entry(call.name.clone())
                .or_insert(0);
            *counter += 1;
            let triggered = *counter >= 3;

            if triggered {
                // Stacking prevention (landmine #11)
                let stacking = state.last_nudge_iteration
                    .map_or(false, |prev| state.iteration.saturating_sub(prev) <= 2);
                if !stacking {
                    crate::brain_planner::reject_plan(last_user_text);
                    state.replans_this_run = state.replans_this_run.saturating_add(1);
                    state.last_nudge_iteration = Some(state.iteration);
                    crate::commands::emit_stream_event(app, "blade_loop_event", serde_json::json!({
                        "kind": "replanning",
                        "count": state.replans_this_run,
                    }));
                    conversation.push(crate::providers::ConversationMessage::System(
                        "Internal check: re-plan from current state. Do not retry the failing step verbatim.".to_string()
                    ));
                }
                *counter = 0;  // reset regardless — we acted on this streak
            }
        }

        let rendered = tool_err.render_for_model();
        (rendered, true)
    }
```
