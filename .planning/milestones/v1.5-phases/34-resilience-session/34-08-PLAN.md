/// SESS-01 — atomic append-only writer. One per session_id.
pub struct SessionWriter {
    pub(crate) path: PathBuf,
    pub(crate) enabled: bool,
}

impl SessionWriter {
    /// SESS-01 — construct a writer for a new conversation. Generates a ULID,
    /// creates jsonl_log_dir if absent, RUNS ROTATION BEFORE creating the new
    /// file (rotation considers existing files only — the new file is not at
    /// risk of being archived on creation).
    ///
    /// When enabled=false, returns a no-op writer (no directory created, no
    /// rotation run). Mirror of Plan 34-03 stub semantics; the difference is
    /// that the real ULID is now generated.
    pub fn new(jsonl_log_dir: &Path, enabled: bool) -> std::io::Result<(Self, String)> {
        let id = Ulid::new().to_string();
        if !enabled {
            return Ok((Self { path: PathBuf::new(), enabled: false }, id));
        }
        std::fs::create_dir_all(jsonl_log_dir)?;
        // Rotation BEFORE the new file is created.
        let cfg = crate::config::load_config();
        rotate_old_sessions(jsonl_log_dir, cfg.session.keep_n_sessions as usize)?;
        let path = jsonl_log_dir.join(format!("{}.jsonl", &id));
        Ok((Self { path, enabled: true }, id))
    }

    /// SESS-01 — no-op writer used by error-recovery paths in commands.rs.
    pub fn no_op() -> Self {
        Self { path: PathBuf::new(), enabled: false }
    }

    /// SESS-01 — atomic append. Wrapped in catch_unwind so a panic in
    /// serialization or I/O does NOT crash the chat. Mirrors Phase 33-09
    /// CTX-07 fallback discipline.
    ///
    /// On Unix, flock(LOCK_EX) protects against concurrent appends from
    /// future multi-window scenarios. On Windows, NTFS append with O_APPEND
    /// is atomic for any size; flock is a no-op there but fs2 wraps both.
    pub fn append(&self, event: &SessionEvent) {
        if !self.enabled {
            return;
        }
        let path = self.path.clone();
        let event_clone = event.clone();
        let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
            #[cfg(test)]
            SESS_FORCE_APPEND_PANIC.with(|p| {
                if p.get() {
                    panic!("test-only induced panic in SessionWriter::append (Plan 34-08 regression)");
                }
            });
            let line = serde_json::to_string(&event_clone)
                .map_err(|e| std::io::Error::new(std::io::ErrorKind::Other, e.to_string()))?;
            let mut f = OpenOptions::new()
                .create(true)
                .append(true)
                .open(&path)?;
            // Advisory lock — protects against multi-writer interleaving.
            f.lock_exclusive()?;
            f.write_all(line.as_bytes())?;
            f.write_all(b"\n")?;
            // Best-effort unlock; even if unlock fails, the file handle close
            // releases the lock. Don't propagate unlock errors.
            let _ = f.unlock();
            Ok::<(), std::io::Error>(())
        }));
        match result {
            Ok(Ok(())) => {}
            Ok(Err(e)) => eprintln!("[SESS-01] append io error at {}: {}", self.path.display(), e),
            Err(_) => eprintln!("[SESS-01] append panicked at {}; chat continues", self.path.display()),
        }
    }
}

/// Wall-clock millis since UNIX epoch. Used by every event emitter for
/// `timestamp_ms`. Returns 0 if the system clock is before 1970 (which
/// shouldn't happen but degrades gracefully).
pub(crate) fn now_ms() -> u64 {
    std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .map(|d| d.as_millis() as u64)
        .unwrap_or(0)
}

/// SESS-01 — rotate old sessions when count > keep_n. Move oldest (by ULID
/// prefix lex sort) to {dir}/archive/. Move, not delete (CONTEXT lock §SESS-01).
pub(crate) fn rotate_old_sessions(dir: &Path, keep_n: usize) -> std::io::Result<()> {
    let mut entries: Vec<PathBuf> = std::fs::read_dir(dir)?
        .filter_map(|e| e.ok())
        .map(|e| e.path())
        .filter(|p| p.extension().and_then(|s| s.to_str()) == Some("jsonl"))
        .collect();
    if entries.len() <= keep_n {
        return Ok(());
    }
    // ULID prefix is lex-sortable by timestamp; oldest first.
    entries.sort();
    let archive_dir = dir.join("archive");
    std::fs::create_dir_all(&archive_dir)?;
    let to_move = entries.len() - keep_n;
    for old in entries.iter().take(to_move) {
        let name = old.file_name().unwrap_or_default();
        let dst = archive_dir.join(name);
        if let Err(e) = std::fs::rename(old, &dst) {
            eprintln!("[SESS-01] rotation failed for {}: {}", old.display(), e);
        }
    }
    Ok(())
}

pub(crate) fn emit_with_jsonl(
    app: &tauri::AppHandle,
    writer: &crate::session::log::SessionWriter,
    kind: &str,
    payload: serde_json::Value,
) {
    // The full event payload merges {kind: ...} + the rest.

        kind: kind.to_string(),
        payload,
        timestamp_ms: crate::session::log::now_ms(),
    });
}
