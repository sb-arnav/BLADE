}

// near other helpers in ChatComposer.tsx
import { AnchorChip } from './AnchorChip';

const ANCHOR_RE = /\B@(?:(screen)\b|file:(\S+)|memory:(\S+))/g;

function renderWithAnchors(text: string, anchorEnabled: boolean): ReactNode[] {
  if (!anchorEnabled) return [text];
  const out: ReactNode[] = [];
  let last = 0;
  for (const match of text.matchAll(ANCHOR_RE)) {
    const [full, screen, filePath, memTopic] = match;
    const idx = match.index ?? 0;
    if (idx > last) out.push(text.slice(last, idx));
    if (screen) {
      out.push(<AnchorChip key={`s-${idx}`} variant="screen" />);
    } else if (filePath) {
      out.push(<AnchorChip key={`f-${idx}`} variant="file" payload={filePath} />);
    } else if (memTopic) {
      out.push(<AnchorChip key={`m-${idx}`} variant="memory" payload={memTopic} />);
    }
    last = idx + full.length;
  }
  if (last < text.length) out.push(text.slice(last));
  return out;
}
