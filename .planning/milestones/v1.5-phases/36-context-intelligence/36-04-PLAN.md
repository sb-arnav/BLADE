//! INTEL-03 budget-bounded repo map + brain.rs injection helpers.

use std::collections::HashSet;

use once_cell::sync::Lazy;
use regex::Regex;
use rusqlite::Connection;

use crate::config::BladeConfig;
use super::pagerank::rank_symbols;
use super::symbol_graph::{SymbolKind, SymbolNode};

const TOKENS_TO_CHARS: u32 = 4;  // chars/4 approximation per Phase 32

static RUST_IDENT_RE: Lazy<Regex> = Lazy::new(|| {
    Regex::new(r"\b[a-z_][a-z0-9_]*(?:::[a-zA-Z_][a-zA-Z0-9_]*)?\b").unwrap()
});
static PASCAL_RE: Lazy<Regex> = Lazy::new(|| {
    Regex::new(r"\b[A-Z][a-zA-Z0-9]*\b").unwrap()
});

pub fn build_repo_map(
    query: &str,
    mentioned_symbols: &[String],
    token_budget: u32,
    config: &BladeConfig,
    conn: &Connection,
) -> Option<String> {
    if !config.intelligence.tree_sitter_enabled { return None; }
    if token_budget == 0 { return None; }

    let damping = config.intelligence.pagerank_damping;
    let ranked = rank_symbols(query, mentioned_symbols, damping, conn);
    if ranked.is_empty() { return None; }

    let rendered = render_map(&ranked, token_budget);
    if rendered.is_empty() { return None; }
    Some(rendered)
}

pub fn harvest_mentioned_symbols(query: &str, recent_messages: &[&str]) -> Vec<String> {
    let mut set: HashSet<String> = HashSet::new();
    let mut ordered: Vec<String> = Vec::new();

    let mut harvest_into = |text: &str, weight: u32| {
        for cap in RUST_IDENT_RE.find_iter(text) {
            let s = cap.as_str().to_string();
            if s.len() < 3 || COMMON_WORDS.contains(&s.to_lowercase().as_str()) { continue; }
            for _ in 0..weight {
                if set.insert(s.clone()) {
                    ordered.push(s.clone());
                }
            }
        }
        for cap in PASCAL_RE.find_iter(text) {
            let s = cap.as_str().to_string();
            if s.len() < 3 { continue; }
            for _ in 0..weight {
                if set.insert(s.clone()) {
                    ordered.push(s.clone());
                }
            }
        }
    };
    harvest_into(query, 2);
    for m in recent_messages.iter().take(10) {
        harvest_into(m, 1);
    }
    ordered
}

pub fn render_map(rows: &[(SymbolNode, f32)], token_budget: u32) -> String {
    let char_budget = token_budget.saturating_mul(TOKENS_TO_CHARS) as usize;
    let header = format!("REPO MAP (top symbols by relevance, ~{} tokens budget):\n", token_budget);
    let mut out = header.clone();
    let mut emitted = 0;
    for (sym, score) in rows {
        let kind_str = match sym.kind {
            SymbolKind::Function => "function",
            SymbolKind::Type => "type",
            SymbolKind::Module => "module",
            SymbolKind::Constant => "constant",
        };
        let line = format!("{}::{} ({}, score={:.3})\n", sym.file_path, sym.name, kind_str, score);
        if out.len() + line.len() > char_budget { break; }
        out.push_str(&line);
        emitted += 1;
    }
    if emitted == 0 { return String::new(); }
    out
}
