Land INTEL-06: @screen / @file: / @memory: anchor parsing + per-anchor injection wired into commands.rs.

`extract_anchors(query: &str) -> (String, Vec<Anchor>)` parses the user query against three regex patterns evaluated in order:
- `\B@screen\b` → `Anchor::Screen`
- `\B@file:([^\s]+)` → `Anchor::File { path: capture }`
- `\B@memory:([^\s]+)` → `Anchor::Memory { topic: capture }`

The `\B` word-boundary discipline is load-bearing — it correctly excludes email addresses like `arnav@pollpe.in` (where `@` follows a word character) from matching `@screen`. Each match is captured, stripped from the query (replaced with single space), and added to the anchors Vec; multiple anchors of the same type are deduped by `(type, payload)`.

`resolve_anchors(anchors, app, config) -> Vec<(label, content)>` routes per-anchor:
- `Anchor::Screen` → existing screenshot OCR path (Phase 32 vision section harvester)
- `Anchor::File { path }` → `std::fs::read_to_string` capped at 200k bytes; binary detection (null-byte in first 8KB) rejects with `[ANCHOR:@file rejected: binary]`
- `Anchor::Memory { topic }` → `memory.rs::query_memories(topic, limit=5)`

Wired into `commands.rs::send_message_stream_inline` at user-query intake (post-input-validation, pre-prompt-build). Behind catch_unwind per CTX-07 v1.1 discipline. When `config.intelligence.context_anchor_enabled = false`, the parser early-returns the original query verbatim.

Anchored content bypasses Phase 32 selective gating — `record_section` labels are `"anchor_screen"`, `"anchor_file"`, `"anchor_memory"` — explicit user asks, not heuristic-driven.

Test coverage: 7 unit tests (3 anchor extractions, 1 fuzz, 1 email word-boundary, 1 force-panic seam, 1 dedup) + 2 integration tests (OCR injects, file truncates).

//! INTEL-06 @screen / @file: / @memory: regex-based anchor extraction.

use std::cell::Cell;
use std::collections::HashSet;

use once_cell::sync::Lazy;
use regex::Regex;
use serde::{Deserialize, Serialize};

use crate::config::BladeConfig;

#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "lowercase")]
pub enum Anchor {
    Screen,
    File { path: String },
    Memory { topic: String },
}

static SCREEN_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"\B@screen\b").unwrap());
static FILE_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"\B@file:([^\s]+)").unwrap());
static MEMORY_RE: Lazy<Regex> = Lazy::new(|| Regex::new(r"\B@memory:([^\s]+)").unwrap());

#[cfg(test)]
thread_local! {
    pub static INTEL_FORCE_ANCHOR_PANIC: Cell<bool> = const { Cell::new(false) };
}

#[cfg(not(test))]
thread_local! {
    pub static INTEL_FORCE_ANCHOR_PANIC: Cell<bool> = const { Cell::new(false) };
}

pub fn extract_anchors(query: &str) -> (String, Vec<Anchor>) {
    if INTEL_FORCE_ANCHOR_PANIC.with(|c| c.get()) {
        panic!("forced anchor parser panic via test seam");
    }

    let mut working = query.to_string();
    let mut anchors: Vec<Anchor> = Vec::new();
    let mut seen: HashSet<(String, String)> = HashSet::new();

    // Pass 1: @screen
    loop {
        let m = match SCREEN_RE.find(&working) { Some(m) => m, None => break };
        let key = ("screen".to_string(), String::new());
        if seen.insert(key) { anchors.push(Anchor::Screen); }
        let r = m.range();
        working.replace_range(r, " ");
    }
    // Pass 2: @file:PATH
    loop {
        let c = match FILE_RE.captures(&working) { Some(c) => c, None => break };
        let m = c.get(0).unwrap();
        let path = c.get(1).unwrap().as_str().to_string();
        let key = ("file".to_string(), path.clone());
        if seen.insert(key) { anchors.push(Anchor::File { path }); }
        let r = m.range();
        working.replace_range(r, " ");
    }
    // Pass 3: @memory:TOPIC
    loop {
        let c = match MEMORY_RE.captures(&working) { Some(c) => c, None => break };
        let m = c.get(0).unwrap();
        let topic = c.get(1).unwrap().as_str().to_string();
        let key = ("memory".to_string(), topic.clone());
        if seen.insert(key) { anchors.push(Anchor::Memory { topic }); }
        let r = m.range();
        working.replace_range(r, " ");
    }

    let cleaned = working.split_whitespace().collect::<Vec<_>>().join(" ").trim().to_string();
    (cleaned, anchors)
}

pub async fn resolve_anchors(
    anchors: &[Anchor],
    app: &tauri::AppHandle,
    _config: &BladeConfig,
) -> Vec<(String, String)> {
    let mut out = Vec::new();
    for a in anchors {
        match a {
            Anchor::Screen => {
                let ocr = current_ocr_text(app).await.unwrap_or_default();
                out.push(("anchor_screen".to_string(),
                    format!("[ANCHOR:@screen]\n{}", crate::safe_slice(&ocr, 8000))));
            }
            Anchor::File { path } => {
                out.push(("anchor_file".to_string(), resolve_file(path)));
            }
            Anchor::Memory { topic } => {
                let facts = crate::memory::query_memories(app, topic, 5).await
                    .unwrap_or_default();
                let joined = facts.iter().take(5).map(|f| f.text.as_str()).collect::<Vec<_>>().join("\n");
                out.push(("anchor_memory".to_string(),
                    format!("[ANCHOR:@memory:{topic}]\n{}", crate::safe_slice(&joined, 4000))));
            }
        }
    }
    out
}

fn resolve_file(path: &str) -> String {
    let p = std::path::PathBuf::from(path);
    if !p.exists() {
        return format!("[ANCHOR:@file:{path} not found]");
    }
    let bytes = match std::fs::read(&p) {
        Ok(b) => b,
        Err(e) => return format!("[ANCHOR:@file:{path} read error: {e}]"),
    };
    if is_binary(&bytes) {
        return format!("[ANCHOR:@file:{path} rejected: binary]");
    }
    let max = 200_000usize;
    let (content, suffix) = if bytes.len() > max {
        (String::from_utf8_lossy(&bytes[..max]).to_string(),
         format!("\n[truncated from {} bytes]", bytes.len()))
    } else {
        (String::from_utf8_lossy(&bytes).to_string(), String::new())
    };
    format!("[ANCHOR:@file:{path}]\n{content}{suffix}")
}

fn is_binary(bytes: &[u8]) -> bool {
    let head = if bytes.len() > 8192 { &bytes[..8192] } else { bytes };
    head.iter().any(|&b| b == 0x00)
}
